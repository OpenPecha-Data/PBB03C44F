ན་མོ་བི་ག་ཏ་རཱ་ག་མཻ་ཏྲེ་ཡ་སིཾ་ཧ་ལོ་ཙ་ནེ་སྭཱ་ཧཱ། 

གོང་བསྙེན་ཁྲི་ཕྲག་བརྒྱད་དེ་ཞོག་རེ་བརྒྱ་རྩ་བརྒྱད་རེ་བཟླས་པའི་སྔགས་ཆུ་མིག་ལ་བྱུག་ན་མིག་མི་འགྲིབ་ཞིང་གསལ་བར་འགྱུར་རོ། །ཟ་ཡིག་ཕྱོགས་བཅུ་མ་ནི། བྷཿནཿདྷཿནཿཀུཿདྷཿནཿཡཿདཿནཿ རྒྱ་ཤོག་ལ་རྒྱ་སྣག་གིས་བྲིས་ལ། ཚེ་སྒྲུབ་དང་བསྲུང་བཟློག་ལ་བདུད་རྩི་ཆོས་སྨན་དང༌། ཡང་ཚེ་སྒྲུབ་ལ་བཟང་དྲུག །རྒྱལ་པོར་གུ་གུལ། ཀླུ་ལ་གླ་རྩི། བཙན་ལ་ཤུ་དག །གཉན་ལ་ཁྱུང་ལྔ། དུག་ལ་རྨ་བྱའི་སྒྲོ་མདོངས། མཚོན་ལ་པདྨ་རཀྟ། ཡམས་ལ་སྤུ་ནག་སོགས། གཉེན་པོ་སོ་སོར་བྱུགས་ནས་ཡར་ལ་སྒྲིལ། གཟའ་ལ་སྤོས་དཀར་བྱུག་ནས་འགོ་ནས་མར་དྲིལ་ལོ། །ཟ་ཡིག་སྤྱི་ཁྱབ་རྫས་ནི་ཆོས་སྨན་ནོ། །དེ་ནས་རབ་གནས་ནི། རང་ཡི་དམ་དུ་བསྒོམ། ནམ་མཁར་ཨོ་རྒྱན་ཡབ་ཡུམ་བསྒོམ། གསོལ་བ་བཏབ། བཛྲ་གུ་རུ་བཟླས། ཛཿཧཱུྃ་བྃ་ཧོཿ ཞེས་པས་ཨོ་རྒྱན་ཡབ་ཡུམ་ཟ་ཡིག་ལ་ཐིམ་པར་བསྒོམ། གོང་དུ་ཞི་དྲག་གི་སྙིང་པོ་བརྒྱ་རྩ་དང༌། ཡེ་དྷརྨས་རབ་གནས་ཀྱང་བྱ་དགོས། ཟ་དུས་མདུན་གྱི་ནམ་མཁར་ཨོ་རྒྱན་བསྒོམ། གནས་གསུམ་ཡིག་འབྲུ་གསུམ་ལས་འོད་འཕྲོས། ཕྱོགས་བཅུའི་ཞིང་ན་སངས་རྒྱས་བྱང་སེམས་སྤྱན་དྲངས་ཨོ་རྒྱན་ལ་ཐིམ། ཨོ་རྒྱན་འོད་ཞུ་ཟ་ཡིག་ལ་ཐིམ། མགོ་ཕྱིར་དང་འཕང་ཤམ་བསྟན་མིད། སྙིང་ཁའི་ཐད་སླེབས་གུ་རུ་དྲག་པོར་གྱུར། སྐུ་ལས་མེ་འབར་ནད་གདོན་བསྲེག་པར་བསྒོམ། ཚེ་སྒྲུབ་སྲུང་བ་ཉི་མ་རྩེ་ཤར་ལ་བྱ། གཞན་རྣམས་སྲོད་ལ་བྱའོ། ༈ ། ཨོཾ་ཐི་མ་ཐ་ར་ཏི་ས་དུན་ཏེ་ཛྭ་ལ་རཾ་པ་ཡེ་སྭཱ་ཧཱ། རང་ཉིད་ཕག་ཏུ་བསྒོམ་ནས་སྣ་བུག་གཡོན་དུ་ཕུས་འདེབས་རིམས་སྲུང་ངོ༌། ། ཨོཾ་པདྨ་ཤ་ཝ་རི་ཕཊ། ནན་པ་ཏ་ དཔེ་གཅིག་ལ་ཏ་ལ་གི་གུ་ཡོད་ ཤིག །ན་ག་ནན། ཏདྱ་ཐཱ། སརྦ་བི་རི་ཏ་ དཔེ་གཉིས་ལ་ཏ་ལ་གི་གུ་ཡོད། ཧ་ན་ཧ་ན་བཛྲ་ན། རཀྵ་རཀྵ་ཡེ་སྭཱ་ཧཱ། རིམས་སྲུང་ནད་བརྒྱ་སྔགས་གཅིག་འདི། དཔེ་ཁུངས་བཙུན་བཞི་ལ་གཏུག་ནས་བྲིས་པ་ཡིན། གོང་བསྙེན་འབུམ་ཕྲག་གསུམ་བཟླས་ཏེ། དུ་བ་སྣ་སྒོར་མ་སོང་གོང་ཉེར་གཅིག་བཟླ་ལ་སྣ་སྒོར་ཕུ་བཏབ་བོ། །གཟའ་དང་དུག་དང་རིམས་ལ་སྔགས་ཆུས་ཕན།

སྤྱིར་ནད་རིགས་བཞི་བརྒྱ་རྩ་བཞི་ལ་ཕན། ཕྱི་མ་འཇའ་ལུས་ཐོབ་པར་བཤད། མི་ལའི་མིག་གི་སྔགས་ནི། ཨོཾ་ལྷང་ཡུལ་ཁང་སྭཱ་ཧཱ། ཏེ་ར་ཝ་ལའི་དྲུག་སྔགས་ནི། ཧྲིཾ་ནི་ལ་ཀུ་ལ་ཁ་ཁྲ་ཁྲ། ལོ་རས་པའི་དུག་སྔགས་ནི། བི་ཤ་འོང༌། བྷི་ཀེ་བྷུན་ཛུ། བྷི་ཤ་ཀེ་རོ་ལི་འོང༌། བཛྲ་བྷ་བྷི་ཤ་ས་འོང༌། ཏི་ནི་བུམ་པ་ན་ར་ཀ་འོང༌། ཨི་ན་ཏི་རི་ཀི་རི། མི་ན་དྷི་ཀི་རི། སཾ་པ་ལོ་ས་ནོ་ས་ཏེ། ཨ་པ་ཚ་ལོང༌། སཾ་པ་ན་ས་ཏ་ལ། བྷ་སྭ་ཀི་ཙ་ལོང༌། ཨན་ཏིར་ཏིར་ཙ་ལོང༌། ཤི་བྷིར་ཨ་ཤོ་རི་བྷིས། གུ་རུ་ལ་བྷུདྡྷོ། ཏ་ཀ་ར། གོང་བསྙེན་སྟོང་ཚོ་གཅིག་སོང་བ་དང༌། ལྟོ་སྟོང་ལ་བདུན་ནམ་ཉེར་གཅིག་བཟླས། ཧ་ཀ་ར། དབུགས་ཐེངས་རེ་ལ་ཐུབ་ཚད་བཟླས། ཐོ་རངས་བྱ་སྐད་མ་ཆོད་པ་གལ་ཆེ། ༈ ཨ་ སྤྲའི་སྔགས་ཡིན་ པར་ཁྲོ་ལོ་ལོ། ཐང་ཆེན་པོའི་དབུས་སུ་ཙནྡན་ཡལ་ག་གསུམ་གྱི་དབུས་སུ་རང་ཉིད་ལྷ་མོའི་ཆ་བྱད་རྐང་གཡས་གཡོན་ཡལ་གའི་སྟེང་ལྷ་མོ་གཉིས་ཀྱི་གླིང་བུ་འབུད་པ་བསྒོམ། དེས་མཐོང་ངོ༌། ༈ །མུ་སྟེགས་རྦད་བཞི་མ་ནི། རང་ཉིད་དབང་ཕྱུག་ཆེན་པོ་སྐུ་མདོག་དུད་ཁ་ཞལ་གཅིག་ཕྱག་གཉིས། གཡས་ཀྱིས་རྩེ་གསུམ་དང༌། གཡོན་ཐོད་ཁྲག་འཛིན་པ། ཡ་མཆུ་ལ་མེ། མ་མཆུ་ལ་རླུང༌། ལྗགས་ལ་མེའི་ཚྭ་ཚྭ་འཕྲོ་བ། རལ་པ་ས་གཞི་ལ་ཁྱབ་པ། ལྟོ་བ་ངན་སྔགས་ཀྱི་རྒྱ་མཚོ་འཁྱིལ་བར་བསྒོམ་ལ། ནག་པོ་རྦད། དུ་ན་ཡ་ དཔེ་གཉིས་ལ་ཡ་ལ་འགྲེང་པོ་མི་འདུག །དཔེ་གཉིས་ལ་ཡ་ལ་འགྲེང་པོ་འདུག་ རྦད། མ་ཧཱ་པཉྩ་རྦད། ཐུན་རིལ་ཛ་རྦད། གོང་བསྙེན་སྟོང་ཕྲག་དྲུག་དེ་ནས་ཞག་རེར་ཉེར་གཅིག་རེ་མ་ཆག་པ་དམ་ཚིག་གོ། །གཉན་ལྷོག་སྐྲང་རིགས་ལ་སྔགས་འདེབས། སོ་སྲིན་ལ་ཚྭ་ལ་སྔགས་ཏེ་སོ་བར་འཛིན། ཚྭ་སྲིན་ལ་ཚྭ་ལ་བཏབ་ཏེ་ཆུ་བཀོལ་དབུལ། གྲང་སྲིན་ཆུ་གྲང་གིས་ཕུལ། ཡ་མ་ལ་ཡ་ཁུང་དུ་འདེབས། ལིང་ཏོག་ལ་ཕུས་འདེབས་ཡལ་བ་བསམ། ཡང་ན་ཕོ་རོག་སྒྲོ་ལ་བཏབ་མིག་ཆུ་ནང་དུ་བལྟས་ལ་སྒྲོས་བྱབ་བོ། །ཤུ་བ་ལ་མར་རྙིང་ལ་བཏབ་ནས་བྱུག་ཉི་མར་བསྲོ།

རྨ་འབྲས་ལ་སྣུམ་རྙིང་པར་འདེབས་སོ། །ཁྲག་ཆོད་ལ་ཐེབ་ཆེན་ལ་སྔགས་ནས་བརླན་ནོ། །རྨ་ལ་ཡང་ཕན་ནོ། །ཡང་གྲོག་ལ་བཏབ་ནས་བརླན་ན་རྨ་ཕྱོགས་མེད་པ་ཕན་ནོ། །སྣུམ་གང་རིགས་ནང་རས་བཅོས་ཏེ་ཁྱི་རྨ་ལ་བླན་ན་ཕན་ནོ། །ཡང་ཁྱི་སྤུ་ལ་སྔགས་ཁྱི་སྤུ་མེར་བསྲེག་པའི་གཞོབ་དེ་མ་ཐག་རྨ་ལ་བརླན། དེ་སྟེང་རེ་བ་ནག་པོས་བཀབ། སྐུད་པ་དཀར་པོས་རྒྱ་གྲམ་དུ་བཅིང་ཞག་གསུམ་ནས་ལྟའོ། །ལྤགས་ནད་ལ་སྒྱེད་པུ་གསུམ་གྱི་དྲེག་ནག་དང་མར་རྙིང་ཞོ་ཤ་རྙིང་པ་ལ་བསྲེས་ཏེ་སྔགས་བཏབ་ལ་ཉི་མ་ལ་བསྲེག་གོ །གྲང་གཉན་ལ་ཆང་ལ་བཏབ་ཁོང་དུ་བཏང་བས་ཕན་ནོ། །སྐྱེས་པའི་དཀར་ཆ་འཛག་པ་ལ། ཁྱི་ནག་པོའི་སྤུ་ཐེ་གུ་གསུམ་སྒྲིལ་ལ་རྡོ་རྗེའི་མདུད་པ་གསུམ་རྒྱབ་ལ། མདུད་པ་རེ་རེ་ལ་སྔགས་སྟོང་ཚོ་རེ་འདེབས། སྐེད་པར་བཅིངས་པས་ཆད་དོ། །འགོས་ནད་བསྲུང་ན་དྲི་ཆུ་ལ་བཏབ་ལྟོ་སྟོང་ལ་བཏང་ངོ༌། །ཐོར་པ་སྲུང་ན་སྲན་མ་ལ་བཏབ་སྐེ་ལ་འདོག་གོ་མཚལ་དཀར་བཀོལ་བ་ལ་སྔགས་ཁྲི་ཚོ་བཏབ་པ་ཁོང་དུ་བཏང་ན་འཕྲལ་དུ་བཤལ་ཡོང༌། ཡུངས་ཀར་ནང་མར་རྙིང་ལ་བཏབ་ན་བྱིས་པ་ཐོན་ནོ། །ཡང་རྒྱུ་གཟེར་ལ་སྔགས་ཆུ། དམར་ནད་ལ་གླ་རྩིའི་ཆུ། ཆམ་པ་ལ་གངས་ཆུར་བཏབ་ནས་སྐར་མ་མ་གྱེས་གོང་བཏང༌། ཞོ་ཉེས་ལ་རྡེའུ་དཀར་ནག་ལ་བཏབ་སྣོད་དུ་བླུག་གོ། །གི་ཝཾ་ལ་བཏབ་ནས་དཔྲལ་བར་ཐིག་ལེ་བྱས་ན་དབང་དུ་འགྱུར་རོ། །དེ་བཞིན་སིནྡྷུ་རས་སྙིང་དུ་སྡུག་གོ །ཡུངས་ཀར་ལ་བཏབ་ནས་ཐུན་གྱིས་བྲབ་ན་འབྱུང་པོ་ཀུན་བྲོས་སོ། །ཐལ་ཆེན་དང་ཡུངས་ཀར་ལ་བཏབ་ནས་གཏོར་ན་སེར་བ་ཐུབ་བོ། །ཡང་ན་རང་དབང་ཕྱུག་ཆེན་པོ་ཐལ་མདོག །རལ་པ་སྟོང་གསུམ་ཁེངས་པ་བསྒོམ། ཡུལ་དང་བསྲུང་བྱ་ཐམས་ཅད་རལ་པའི་འོག་ཏུ་མནན་པར་བསམས་ཏེ་སྡིགས་མཛུབ་བྱས་པས་ཐུབ་བོ། །སོར་མོ་ལྔ་ལ་བཏབ་ནས་ཁུ་ཚུར་བཅང་ནས་ཕྱིན་ན་སུའི་རྗེས་ཀྱང་ཟིན་ནོ། ། ༈ ཨ་ཊེ་བ་ཊེ་ན་ཊེ་ཀུ་ན་ཊེ་ཊ་ཀེ་ཋ་ཀེ་ཐ་ར་ཀེ་ཨུ་རུ་མ་ཏི་རུ་རུ་མ་ཏི་རུ་རུ་ཧི་ལི་མི་ལི་སཔྤ་ཛྙོ་དུ་པ་དགྒ་ན་མོ་སཔྤ་སམྨཱ་སི་བུཛྫྙཱ་ཎཾ་ཛྫྙནྟུ་མེ་མནྟྲ་པ་དཱཿསྭཱ་ཧཱ།

སྒྲ་མི་སྙན་གྱི་ཚེ་གཟུངས་སོ། ༈ ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སྭཱ་ཧཱ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་མ་མ་ཨཱ་ཡུཿཔུཎྱ་ཛྙཱ་ན་པུཥྚིཾ་ཀུ་རུ་སྭཱ་ཧཱ། སྒྲོལ་མ་དཀར་མོའི་སྙིང་པོའོ། ༈ །དབང་པོ་རྟུལ་བའི་རིགས་ལ། བཅོམ་ལྡན་འདས་ཀྱི་ལམ་བར་གནང་བའི་ཚིགས་འདི་གཉིས། རྡུལ་སྤང་ངོ༌། །དྲི་མ་སྤང་ངོ༌། ། ཞེས་བརྗོད་པས་སྔོན་ཚིག་གཅིག་མི་ཤེས་ཀྱང་ཕྱིས་མཁས་པ་དགྲ་བཅོམ་པ་ཆེན་པོར་གྱུར་པའོ། ༈ །དམ་ཆོས་མཐར་ཕྱིན་གྱིས། སྟོང་ཉིད་ཀྱི་ངང་ནས་ནཱ་ག་རཀྵའི་སྔགས། ཨོཾ་ལཾ་ལཾ་ཧཱུྃ། རྐྱང་བ་བཏོན་པས་ཀླུ་གདོན་ལས་གྲོལ་བ་ཡིན་གསུངས་སོ། ། ཨོཾ་མཉྫུ་ཤྲཱི་ཡ་མནྟ་ཀཿ ཧ་ཡ་གྲཱྀ་ཝཿ བཛྲ་པཱ་ཎིཿ དུ་ཏྲི་ཡ་མ་ར་ཙཿ ཡ་མ་རཀྵཿ ཀྲོ་དྷ་ས་ཙ་མཿ ཀཱ་ལ་རུ་པཿ ཨ་ཀྲོ་ར་ཙཿ ཡ་མ་ར་ཙཿ ས་མ་ཡ་ཛ་ཛཿ བདག་དང་བསྲུང་བྱ་རྣམས་ལ་སྲུངས་ཤིག་ནྲྀ་བྷྲཱུྃ་བཛྲ་རཀྵ་རཀྵཿ ༈ ཟློག་སྔགས་ནི༔ དུ་ཏྲི་ཡ་མ་ར་ཙཿ ཡ་མ་རཀྵཿ ཀྲོ་ཏ་པ་ཙ་ལ་ཏིཿ ཨ་ཡ་མ་དུ་རུ་ཙ་ཤ་ནཿ གནོད་བྱེད་བྱད་མ་ལྷ་སྲུང་དང་བཅས་པ་མཱ་ར་ཡ་བྷྱོཿབྷྱོཿ བཟློག་བཟློགཿ དགེའོ།། །།
