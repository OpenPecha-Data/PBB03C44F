༄༅། །རྒྱུན་ཁྱེར་གྱི་རིམ་པ་རྣམས་ཕྱོགས་གཅིག་ཏུ་བསྡེབས་པ་དངོས་གྲུབ་བདུད་རྩིའི་གཏེར་མཛོད་ཅེས་བྱ་བ་བཞུགས་སོ། །

རང་ཉིད་ཐ་མལ་སྤྱི་བོ་རུ། །ཧཱུྃ་དཀར་ལས་བྱུང་པད་ཟླའི་སྟེང༌། །རྡོར་སེམས་དཀར་པོ་ཕྱག་གཉིས་ཀྱི། །གཡས་པས་རྡོ་རྗེ་གཡོན་དྲིལ་བསྣམས། །སྐྱིལ་ཀྲུང་ལོངས་སྤྱོད་རྫོགས་སྐུའི་ཆས། །སྣང་ལ་རང་བཞིན་མེད་པར་གྱུར། །སྤྱི་མགྲིན་སྙིང་གར་ཨོཾ་ཨཱཿཧཱུྃ། །ཧཱུྃ་ལས་འོད་འཕྲོས་ཡེ་ཤེས་སེམས། །སྤྱན་དྲངས་དབྱེར་མེད་གཅིག་འདྲེས་གྱུར། །ཐུགས་ཀར་ཟླ་སྟེང་ཧཱུྃ་དཀར་མཐར། །ཡི་གེ་བརྒྱ་པ་གཡས་བསྐོར་གསལ། །ཨོཾ་བཛྲ་སཏྭ་ སོགས་བརྗོད། ཕྱི་ནང་སྐུ་གསུང་ཐུགས་ལ་སོགས། །རྩ་བ་ཡན་ལག་དམ་འགལ་རྣམས། །གནོང་འགྱོད་དྲག་པོས་བཟོད་པར་གསོལ། །སྐུ་ཡི་ཆ་ལས་བདུད་རྩི་བབས། །སྒྲིབ་གཉིས་བག་ཆགས་དུད་ཁུ་ལྟར། །ནག་ཉིལ་ཕྱིར་ཐོན་ལྷ་འོད་ཞུ། །རང་དང་དབྱེར་མེད་ཆུ་བཞག་ལྟར། །བདེ་གསལ་མི་རྟོག་ངང་ལ་བཞག །དགེ་བ་འདི་ཡིས་ སོགས། ཀརྨ་མངོན་དགའི་ངོར་སྤེལ།། །། ༈ ཧྲཱིཿ གཟུང་འཛིན་ཆོས་རྣམས་སྟོང་པའི་ངང་ཉིད་ལས། །རང་ཉིད་ཐ་མལ་སྤྱི་བོར་པད་ཟླའི་སྟེང༌། །ཧྲཱིཿདཀར་འོད་ཀྱི་འཕྲོ་འདུས་དོན་གཉིས་བྱས། །ཡོངས་གྱུར་རིགས་འདུས་ཁྱབ་བདག་ཧེ་རུ་ཀ །བཛྲ་ས་ཏྭ་དུང་དང་ཀུནྡའི་མདོག །གསལ་ཞིང་འཚེར་ལ་འོད་དང་འོད་ཟེར་འབར། །དབུ་ལ་རིགས་ལྔའི་པྲོག་ཞུ་དར་དཔྱངས་བརྒྱན། །འགྱིང་བག་ཚུལ་གྱིས་རྡོ་རྗེ་དྲིལ་བུ་འཛིན། །ལྷ་རྫས་དར་གྱི་སྟོད་གཡོགས་སྨད་གཡོགས་བཀླུབས། །རིན་ཆེན་སྙན་ཆ་མགུལ་རྒྱན་ལ་སོགས་པ། །ལོངས་སྐུའི་ཆས་རྫོགས་ཞབས་གཉིས་སེམས་སྐྱིལ་བཞུགས། །གཞན་ཡང་རྩ་གསུམ་ལྷ་ཚོགས་སྤྲིན་ལྟར་གཏིབས། །གནས་གསུམ་འབྲུ་གསུམ་ཐུགས་ཀྱི་ཧཱུྃ་འོད་ཀྱིས། །བསྐུལ་བས་སངས་རྒྱས་བྱང་སེམས་སྤྱན་དྲངས་ཐིམ། །རྒྱལ་བ་རབ་འབྱམས་དཀྱིལ་འཁོར་ཀུན་འདུས་པའི། །བདག་ཉིད་ཆེན་པོར་མངོན་སུམ་ལྷམ་མེར་གསལ། །བདེར་གཤེགས་འདུས་པ་བླ་མ་ཧེ་རུ་ཀ །ཆོས་ཀུན་འབྱུང་གནས་གསང་སྔགས་སྟོན་པ་ཆེ། །ཚོགས་མཆོག་རང་བཞིན་དཔལ་ལྡན་རྡོ་རྗེ་འཛིན། །གསོལ་འདེབས་སྐལ་ལྡན་རྒྱུད་བྱིན་བརླབ་དུ་གསོལ། །

ཅེས་པའི་གསོལ་འདེབས་ལན་ཁ་ཤས་དང༌། བླ་མའི་ཐུགས་ཀར་ཟླ་སྟེང་ཧཱུྃ་མཐར་སྔགས། །ཡིག་བརྒྱས་བསྐོར་བ་དེ་ལས་འོད་འཕྲོས་པས། །འཕགས་མཆོད་འགྲོ་བའི་སྒྲིབ་སྦྱོང་ཁྱད་པར་དུ། །སྐུ་ལས་བདུད་རྩིའི་རྒྱུན་བབས་བདག་ལུས་གང༌། །ཐོག་མེད་ནས་བསགས་སྡིག་སྒྲིབ་བྱང་ཞིང་དག །རྣམ་སྨིན་ལུས་སྦྱོང་དྲི་མེད་ཤེལ་ལྟར་གྱུར། ། དེ་ནས་ཡིག་བརྒྱ་ཅི་ནུས་བཟླས་ཤིང༌། ཡིག་དྲུག་ཀྱང་བཟླའོ། །དེ་ནས་སྤྲོ་ན་བརྗོད་མེད་དོན་བཤགས་བཏོན་ལ་རྒྱལ་གསུང་ཚད་མ་ཉིད་དེ་ལྷག་པར་ཕན་ཡོན་ཆེ་ཞིང༌། བསྡུས་པར་ན། ཀྱཻ་ཧོ་རྒྱལ་ཀུན་འདུས་པ་དགོངས། །བདག་ཅག་མ་རིག་ནོངས་པ་ཡིས། །སོ་ཐར་བྱང་སེམས་གསང་སྔགས་ཀྱི། །དམ་ཚིག་ཉེས་ལྟུང་འགལ་འཁྲུལ་རྣམས། །མི་འཆབ་མི་སྦེད་མཐོལ་ཞིང་བཤགས། །ཕྱིན་ཆད་སྡོམ་ལ་གནས་པར་བགྱི། །མགོན་པོའི་གསུང་ལས་བུ་ཁྱོད་ཀྱི། །སྡིག་སྒྲིབ་དག་ཅེས་དབུགས་དབྱུང་གྱུར། ། རྗེས་རིམ་ནི། ལྷ་སྐུ་འོད་ཞུ་རང་ཐིམ་གཉིས་མེད་ངང༌། །སྤྲོས་བྲལ་བདེ་བ་ཆེན་པོའི་ཀློང་དུ་བཞག ། མཐར་དགེ་རྩ་བྱང་ཆུབ་ཏུ་བསྔོ་བ་སྤྱི་ལྟར་བྱའོ། །ཅེས་པའང་ཀརྨ་ཆགས་མེད་དངོས་སྤྲུལ་ཀརྨ་བསྟན་འཛིན་ཕྲིན་ལས་བཀའི་བསྐུལ་ལྟར་སི་ཏུས་སྤེལ།། །། ༈ འཇམ་དཔལ་དམར་སེར་ཤེས་རབ་བློ་འཕེལ་བཞུགས། ན་མཿསརྦ་ཛྙ་དྷརྨཱ་ཀཱ་ར་ཡ། འདིར་བཅོམ་ལྡན་འདས་འཇམ་དཔལ་དཔའ་བོ་དམར་སེར་གྱི་སྒོ་ནས་ཤེས་རབ་བློ་འཕེལ་སྒྲུབ་པར་འདོད་པའི་གང་ཟག་རྣམས་ལ་ཕན་བདེའི་དཔྱིད་དུ་འདི་ལྟར་སྤེལ་བར་བྱ་སྟེ། བསླུ་མེད་སྐྱབས་གནས་མཆོག་གསུམ་ལ། །བྱང་ཆུབ་བར་དུ་སྐྱབས་སུ་མཆི། །བདག་གིས་ཤེས་རབ་བསླབ་པ་ཡིས། །ཀུན་ཀྱང་ཐམས་ཅད་མཁྱེན་ཐོབ་ཤོག །ཨོཾ་སྭ་བྷཱ་ཝ་ སོགས་ཀྱིས་སྟོང་པར་སྦྱངས། ཆོས་རྣམས་དམིགས་མེད་སྟོང་ཆེན་ངང༌། །སྣ་ཚོགས་པདྨ་འདབ་རྒྱས་དབུས། །ཨ་ལས་བསིལ་ཟེར་བྱེད་པ་ཡི། །དཀྱིལ་འཁོར་དྲི་མེད་ཉ་གང་སྟེང༌། །རང་རིག་ངོ་བོ་དྷཱིཿདམར་སེར། །འོད་འཕྲོས་རྒྱལ་ཀུན་མཆོད་པས་མཉེས། །འགྲོ་བའི་སྒྲིབ་ཚོགས་མུན་པ་བསལ། །སླར་ཡང་འོད་ཟེར་འདུས་པ་ལས། །རང་ཉིད་རྗེ་བཙུན་འཇམ་དཔལ་དབྱངས། །ཉིན་བྱེད་འཆར་ཀའི་མདངས་ཟེར་ལས། །ཆེས་ཆེར་ལྷག་པའི་ཁ་དོག་འཆང༌། །རལ་བའི་ཅོད་པན་སྤྱི་གཙུག་ཏུ། །བཅིང་ཞིང་ཟུར་ཕུད་གཡོན་ངོས་འཕྱངས། །ཞལ་གཅིག་ཕྱག་གཉིས་ཞི་འཛུམ་བརྗིད། །ཕྱག་གཡས་ཤེས་རབ་རལ་གྲི་མཆོག

སོགས་ཀྱིས་མཆོད། སྐྱོན་གྱིས་མ་གོས་ སོགས་ཀྱིས་བསྟོད། རང་གི་སྙིང་གར་ཟླ་སྟེང་ཧྲཱིཿ །མཐའ་མར་ཡིག་དྲུག་གཡས་སུ་འཁོར། །འོད་འཕྲོས་རྒྱལ་བའི་བྱིན་རླབས་བསྡུས། །འགྲོ་བའི་སྡིག་སྒྲིབ་ཀུན་སྦྱང་གྱུར། ། ཡིག་དྲུག་ཅི་ནུས་བཟླས་ལ། ཐུན་མཚམས་སུ་གཏོར་འབུལ་སྤྲོ་ན། ཨ་མྲྀཏ ས་བསངས། སྭ་བྷཱ་ཝ ས་སྦྱངས། བྷཱུྃ་ལས་རིན་ཆེན་སྣོད་དུ་ཨོཾ་འོད་ཞུ། །གཏོར་མ་བདུད་རྩིའི་རང་བཞིན་ཟག་མེད་གྱུར། །རང་གི་སྙིང་གའི་འོད་ཀྱིས་འཇིག་རྟེན་མགོན། །སྤྱན་དྲངས་མཆོད་འབུལ་བདག་རྒྱུད་སྨིན་གྲོལ་མཛོད། །ཨ་ཀཱ་རོ་ ཚར་གསུམ་གྱིས་གཏོར་མ་འབུལ། རང་ཉིད་ལྷར་གསལ་འོད་ཞུ་ནས། །ཐུགས་ཀའི་ཧྲཱིཿཡིག་ལ་ཐིམ་གྱུར། །དེ་ཡི་ངང་ལ་མཉམ་པར་བཞག །སླར་ཡང་སྐད་ཅིག་དྲན་རྫོགས་སུ། །སྤྱན་རས་གཟིགས་དབང་གསལ་བའི་ངང༌། །ཁྱེར་སོ་གསུམ་ལྡན་སྤྱོད་པ་བསྐྱང༌། ། མཐར་བསྔོ་སྨོན་གྱིས་རྒྱས་གདབ་བྱའོ། །ཡོ་ཆུང་མའི་ངོར་སྤེལ།། །། ༈ བཅོམ་ལྡན་འདས་འོད་དཔག་མེད་ཀྱི་སྒོམ་བཟླས་ཉུང་འདུས་སུ་བཀོད་པ་བཞུགས་སོ། །སྭསྟི། འདིར་བཅོམ་ལྡན་འདས་འོད་དཔག་མེད་ལ་བརྟེན་ནས་རང་གཞན་གྱི་དོན་བྱ་བར་འདོད་ན། སངས་རྒྱས་ཆོས་ཚོགས་དང་ཚད་མེད་བཞི་བཅས་སྔོན་དུ་བཏང་ནས། སྭ་བྷཱ་ཝ་ སོགས། སྣང་ཞིང་སྲིད་པའི་ཆོས་རྣམས་ཀུན། །གཟུང་འཛིན་བྲལ་བའི་སྟོང་བར་གྱུར། །དེ་ཡི་ངང་ལས་རང་རྩལ་མདངས། །རྨ་བྱས་བཏེགས་པའི་རིན་ཆེན་ཁྲིར། །སྣ་ཚོགས་པདྨ་ཟླ་བའི་སྟེང༌། །རང་སེམས་ཧྲཱིཿཡིག་འོད་འཕྲོ་འདུས། །དོན་གཉིས་བྱས་ཏེ་ཡོངས་གྱུར་ལས། །བཅོམ་ལྡན་འོད་མི་འགྱུར་བའི་སྐུ། །པདྨ་རཱ་གའི་རི་བོ་ལྟར། །དམར་གསལ་ཞལ་གཅིག་ཕྱག་མཉམ་བཞག །ལྷུང་བཟེད་བདུད་རྩིས་བཀང་བ་འཛིན། །ཆོས་གོས་རྣམ་གསུམ་ཞབས་རྡོར་སྐྱིལ། །མཚན་དཔེའི་འོད་ཀྱི་དཀྱིལ་འཁོར་གསལ། །གཡས་གཡོན་པདྨ་ཟླ་བའི་སྟེང༌། །ཕྱག་ན་པདྨ་རྡོ་རྗེ་འཛིན། །རྒྱལ་སྲས་ཐུ་བོ་མཆོག་བརྒྱད་དང༌། །གཞན་ཡང་སངས་རྒྱས་བྱང་ཆུབ་སེམས། །ཉན་ཐོས་རང་རྒྱལ་ཚོགས་ཀྱིས་བསྐོར། །གཙོ་བོའི་གནས་གསུམ་ཨོཾ་ཨཱ་ཧཱུྃ། །ཐུགས་ཀར་པད་ཟླར་ཧྲཱིཿཡིག་ལས། །འོད་འཕྲོས་བདེ་ཆེན་ཞིང་ཁམས་ནས། །བསྒོམ་པ་དང་འདྲའི་ཡེ་ཤེས་པ། །སྤྱན་དྲངས་དམ་ཡེ་གཉིས་མེད་གྱུར། །སླར་ཡང་འོད་འཕྲོས་དབང་ལྷ་བསྐུལ། །དབང་བསྐུར་སྐུ་གང་དྲི་མ་དག

སྙིང་པོ་ཅི་ནུས་སུ་བཟླས་ལ། ཉེ་སྙིང༌། ཧྲཱིཿ ཅེས་པའང་བཟླས་ལ། མཐར། ཕྱི་སྣོད་ལྷ་ཡི་དཀྱིལ་འཁོར་པ། །རང་ལ་ཐིམ་ཞིང་རང་ཉིད་ཀྱང༌། །ཐུགས་ཀའི་ས་བོན་ལ་ཐིམ་ནས། །དེའང་མི་དམིགས་འོད་གསལ་བ། །མ་བཅོས་སྤྲོས་བྲལ་ངང་ལ་བཞག ། ཅེས་ཅི་གནས་རང་བབས་སུ་མཉམ་བཞག་གི་ངོ་བོ་བསྐྱངས་ལ། སླར་ཡང་ལྷ་སྐུར་གསལ་བ་ཡི། །གནས་གསུམ་འབྲུ་གསུམ་མཚན་པར་གྱུར། །སྣོད་བཅུད་ལྷ་ཡི་རང་བཞིན་སྐུ། །སྒྲ་གྲགས་སྔགས་ལ་དྲན་རྟོག་ཐུགས། །ཀུན་བཟང་ཀློང་ཡངས་ཆེན་པོར་གསལ། །དགེ་བ་འདི་ཡིས་ སོགས་བརྗོད། སྐབས་སུ་གཏོར་མ་འབུལ་བར་སྤྲོ་ན། ཨ་མྲྀ་ཏ ས་བསངས། སྭ་བྷཱ་ཝ ས་སྦྱངས། སྟོང་ལས་རིན་ཆེན་སྣོད་མཆོག་ཏུ། །ལྷ་ཡི་ཞལ་ཟས་གཙང་ཞིང་བཟང༌། །འདོད་ཡོན་ལྔ་ལྡན་མཆོད་པའི་ཚོགས། །རྒྱ་ཆེན་རྒྱུན་མི་ཆད་པར་མོས། །རང་གི་སྙིང་གའི་འོད་ཟེར་གྱིས། །བསྐུལ་ལས་ཡེ་ཤེས་སྤྱན་དྲངས་གྱུར། །ཨོཾ་ཨ་ཀཱ་རོ ས་ལན་གསུམ་བྱིན་བརླབས་ལ་འབུལ། ཨོཾ་བཛྲ་ཨརྒྷཾ་ སོགས་ཀྱིས་མཆོད། གོང་བཞིན་བསྟོད་པ་དང༌། འདོད་དོན་གསོལ་བ་ནི། མཆོད་སྦྱིན་གཏོར་མ་འདི་བཞེས་ལ། ། སོགས་ནས། ཀུན་འགྲུབ་མཛོད་ བར་བརྗོད། མ་འབྱོར་པ་ སོགས་ཀྱིས་བཟོད་གསོལ་དང༌། བཛྲ་མཱུ ས་ཡེ་ཤེས་གཤེགས་པའམ། སུ་པྲ་ཏིཥྛ ས་བརྟན་བཞུགས་བྱ་བའང་ལེགས། ཅེས་པའང་འཇང་ཡུལ་གྱི་སྒྲུབ་པ་ཀརྨ་ཐུགས་རྗེས་བསྐུལ་ལྟར་སི་ཏུ་པས་སྤེལ་བ་མངྒ་ལཾ།། །། ༈ ཞིང་སྒྲུབ་དང་ཚེ་སྒྲུབ་ཉུང་བསྡུས་དངོས་གྲུབ་ཆུ་འཛིན་བི་ཧ་ར་ཏིསྨ། མཆོག་གསུམ་རྒྱལ་ལ་སྐྱབས་མཆི་ཞིང༌། །འགྲོ་དོན་བྱང་ཆུབ་སེམས་བསྐྱེད་དོ། །རང་སེམས་ཧྲཱིཿཡིག་ཡོངས་གྱུར་ལས། །སྤྱན་རས་གཟིགས་དབང་སྐུ་མདོག་དཀར། །ཞལ་གཅིག་ཕྱག་གཉིས་གཡས་མཆོག་སྦྱིན། །གཡོན་པས་པད་དཀར་ཞབས་སྐྱིལ་ཀྲུང༌། །ལོངས་སྤྱོད་རྫོགས་པའི་ཆ་ལུགས་ཅན། །གནས་གསུམ་འབྲུ་གསུམ་མཚན་པར་གྱུར། །མདུན་མཁར་པད་ཟླའི་སྟེང་དུ་ཧྲཱིཿ །ཡོངས་གྱུར་བདེ་ཆེན་པདྨོ་བཀོད། །ཞིང་མགོན་སངས་རྒྱས་འོད་དཔག་མེད། །དམར་གསལ་ཞལ་གཅིག་ཕྱག་གཉིས་ཀྱིས། །མཉམ་བཞག་ལྷུང་བཟེད་སྤྲུལ་པའི་སྐུ། །མཚན་དཔེ་གཟི་འབར་ཆས་རྫོགས་གསལ། །སྐུ་གསུམ་ངོ་བོ་འབྲུ་གསུམ་མཚན། །སླར་ཡང་འོད་འཕྲོས་རྣམ་དག་ཞིང༌། །རྒྱ་མཚོར་མངའ་བའི་བཅོམ་ལྡན་འདས། །ལྷ་ཚོགས་འཁོར་བཅས་ཐུགས་རྒྱུད་བསྐུལ། །སྤྱན་དྲངས་བདག་མདུན་ལ་ཐིམ་གྱུར། །ཨོཾ་ཨ་མི་དྷེ་ཝ་ཛྙཱ་ན་སརྦ་ཨཱ་བེ་ཤ་ཡ་ཧཱུྃ།

ཛཿཧཱུྃ་བྃ་ཧོཿ་གཉིས་མེད་གྱུར། །བདག་མདུན་ཐུགས་ཀའི་ཧཱུྃ་ཡིག་ལས། །མཆོད་པའི་ཚོགས་སྤྲོས་དགྱེས་བསྐྱེད་གྱུར། །དབང་ལྷས་དབང་བསྐུར་སྐུ་ཁྲུས་གསོལ། །ཆུ་ལྷག་ཡར་བརྟེན་རིགས་ལྔ་ཡིས། །རྒྱས་གདབ་ཐུགས་ཀར་ཟླ་སྟེང་ཧྲཱིཿ །དེ་མཐར་སྔགས་བསྐོར་ཕྲེང་བ་ལས། །འོད་འཕྲོས་འཕགས་མཆོད་འགྲོ་སྒྲིབ་སྦྱངས། །བདག་རྒྱུད་ཡེ་ཤེས་སྨིན་གྲོལ་གྱུར། ། དེ་ནས་ལྷ་ཡི་རིགས་སྔགས་བཟླས་ལ། སླར་ཡང་མདུན་བསྐྱེད་ཐུགས་ལས་འོད། །ལྔ་ལྡན་ཕྱོགས་བཅུར་འཕྲོས་པ་ལས། །ཚེ་ཉམས་ཡལ་བ་ཆད་པ་སོགས། །ཐམས་ཅད་ཐིག་ལེ་འོད་ཟེར་དང༌། །ནྲྀ་ཡིག་རྣམ་པར་བསྡུས་ནས་སུ། །རང་ཐིམ་ཚེ་དབང་དཔལ་འབྱོར་པ། །རབ་རྒྱས་འཆི་མེད་མཆོག་ཐོབ་གྱུར། ། སྐབས་འདིར། ཨོཾ་ཨཱཿཧཱུྃ་ཨ་མི་དྷེ་ཝ་ཨཱ་ཡུཿསིདྡྷི་ཧཱུྃ་ཧྲཱིཿ ཅི་ནུས་བཟླས་མཐར། མདུན་བསྐྱེད་འོད་ཞུ་དགྱེས་མདངས་བཅས། །རང་གི་སྤྱི་བོར་ཐིམ་ནས་སུ། །དབྱེར་མེད་གཅིག་འདྲེས་མ་བཅོས་པ། །རང་རིག་རང་གསལ་ཆོས་དབྱིངས་ཀྱི། །རྩལ་གདངས་ཀློང་ཡངས་ཆེན་པོའི་ངང༌། །འཁོར་འདས་དཀྱིལ་འཁོ༷ར་གཅིག་ཏུ་ཤར། ། དར་ཅིག་ལྟ་བའི་ངང་ལ་མཉམ་པར་བཞག །མཐར་དགེ་བསྔོ་སོགས་སྤྱི་འགྲོ་ལྟར་བྱའོ། །ཞེས་པའང་ཞེ་འཛོང་བླ་སྤྲུལ་ཀརྨ་ཨ་མཱ་རའི་བཞེད་ངོར་སྤེལ་བ་ཛ་ག་ཏཾ། བཀྲ་ཤིས་དཔལ་འབར་འཛམ་གླིང་རྒྱན་དྲུག་ཤོག །དགེའོ། །དགེའོ། །དགེའོ།། །། ༈ རྒྱལ་བ་རྒྱ་མཚོའི་ལྷ་དགུའི་བདག་བསྐྱེད་རྒྱུན་ཁྱེར་ངེས་ལམ་སྙིང་པོའི་དོན་སྟོན་བཞུགས་སོ། །ན་མོ་ལོ་ཀི་ཤྭ་ར་ཡེ། འདིར་དཔལ་འཇིག་རྟེན་དབང་ཕྱུག་གི་བདག་བསྐྱེད་རྒྱུན་ཁྱེར་འདོད་པས་མཆོད་པ་དངོས་བཤམས་སམ། ཡིད་སྤྲུལ་དམིགས་པས་ཁྱབ་པར་བྱས་ལ། དང་པོ་སྐྱབས་འགྲོ་སེམས་བསྐྱེད་ཚད་མེད་བཞི་བསྒོམ་པ་ནི། བདག་དང་མ་གྱུར་མཁའ་ཁྱབ་འགྲོ་བ་རྣམས། །ཕྱོགས་བཅུ་དུས་གསུམ་སངས་རྒྱས་ངོ་བོ་ཉིད། །དཀོན་མཆོག་གསུམ་དང་རྩ་གསུམ་འདུས་པའི་དཔལ། །རྒྱལ་བ་རྒྱ་མཚོའི་ལྷ་ལ་སྐྱབས་སུ་མཆི། །འཁོར་བ་དུཿཁའི་མུན་པར་བྱིངས་པ་རྣམས། །དེང་འདིར་བདག་གི་སེམས་བསྐྱེད་འོད་སྣང་གིས། །མྱ་ངན་འདས་པའི་ལམ་ལ་འཇུག་པ་ཡི། །ཡང་དག་ཡེ་ཤེས་ཕྱིར་མི་ལྡོག་ཕྱིར་བསྔོ། །སེམས་ཅན་ཐམས་ཅད་བདེ་ལྡན་སྡུག་བསྔལ་བྲལ། །བདེ་དང་མི་འབྲལ་མཉམ་ཉིད་ངང་གནས་ཤོག

། དེ་ནས་རྟེན་དང་བརྟེན་པའི་དཀྱིལ་འཁོར་བསྒོམ་པ་ནི། ཨོཾ་སྭ་བྷཱ་ཝ ས་སྟོང་པར་སྦྱངས། ཆོས་རྣམས་ཐམས་ཅད་སྟོང་པ་ཉིད་དུ་གྱུར། །དེ་ཡི་ངང་ལས་རང་སེམས་ཡི་གེ་ཧྲཱིཿ། །དམར་གསལ་ཡེ་ཤེས་ལྔ་ལྡན་འོད་ལྔའི་ཟེར། །འོག་གི་ཕྱོགས་འཕྲོས་འོད་ཟེར་ལས་གྲུབ་པའི། །ཞིང་ཁམས་རྒྱ་ཁྱོན་ཆོས་དབྱིངས་ཚད་མཉམ་དབུས། །སྣ་ཚོགས་རྡོ་རྗེའི་ལྟེ་བར་བྱང་ཆུབ་ཤིང༌། །ལོ་འབྲས་ནོར་བུ་རིན་ཆེན་བཀྲ་ཤིས་བརྒྱན། །དེ་སྟེང་པྃ་ལས་པདྨ་འདབ་མ་འབུམ། ལྟེ་བར་བྷྲཱུྃ་ལས་གཞལ་ཡས་ཁང་གྲུ་བཞི། །སྒོ་བཞི་རྟ་བབས་བཞི་ལྡན་ཀ་བརྒྱད་སོགས། །མཚན་ཉིད་ཡོངས་རྫོགས་དཀྱིལ་འཁོར་ཆེན་པོར་གསལ། །ཕྱི་རོལ་དུར་ཁྲོད་བརྒྱད་དང་མཐའ་སྐོར་དུ། །རྡོ་རྗེ་ར་གུར་མེ་རི་འབར་པའོ། །འོད་ཟེར་ཚུར་འདུས་རང་ཐིམ་ཡོངས་གྱུར་ལས། །བཅོམ་ལྡན་སྤྱན་རས་གཟིགས་དབང་སྐུ་མདོག་དམར། །ཞལ་གཅིག་ཕྱག་བཞི་རྩ་ཕྱག་ཐལ་སྦྱར་གྱིས། །ཡུམ་འཁྱུད་གཡས་འོག་རྡོ་རྗེ་རྩེ་ལྔ་དང༌། །གཡོན་འོག་པདྨ་དམར་པོ་འདབ་བརྒྱད་འཛིན། །དབུ་སྐྲ་ཐོར་ཚུགས་དར་གྱི་ཅོད་པན་དང༌། །རིན་ཆེན་རྩེ་བྲན་ལྔས་བརྒྱན་ཞབས་རྡོར་སྐྱིལ། །པང་དུ་ཡུམ་མཆོག་རྡོ་རྗེ་ཕག་མོ་དམར། །ཞལ་གཉིས་སྤྱི་གཙུག་ཕག་ཞལ་ཕེཾ་སྒྲ་ངུར། །རྩ་ཞལ་ཁྲོ་གཉེར་སྤྱན་གསུམ་མཆེ་བ་གཙིགས༌། །ཕྱག་གཉིས་གཡས་པ་རྡོ་རྗེ་གྲི་གུག་འཕྱར། །གཡོན་ཐོད་ཁྲག་བཀང་ཡབ་ལ་སྟོབ་བཞིན་འཁྱུད། །སྐྲ་གྲོལ་ཐོད་སྐམ་ལྔས་བརྒྱན་ཐོད་རློན་གྱི། །དོ་ཤལ་རུས་པའི་ཕྱག་རྒྱ་ལྔས་བརྒྱན་ཞིང༌། །ཞབས་གཉིས་ཐབས་ཀྱི་སྐུ་ལ་འཁྲིལ་ཚུལ་གྱིས། །དམར་གསལ་སྤྱན་མ་སྔོན་མོ་གཉིས་མེད་སྦྱོར། །ལྷོ་རུ་རིན་ཆེན་སྤྱན་རས་གཟིགས་དམར་པོ། །མཱ་ཀི་སེར་མོ་མཉམ་སྦྱོར་ནུབ་ཏུ་ནི། །སྣང་མཐའ་སྤྱན་རས་གཟིགས་དམར་གོས་དཀར་མོ། །དཀར་དམར་མདངས་ཅན་མཉམ་སྦྱོར་བྱང་དོན་གྲུབ། ཞེས་རྡོ་རྗེ་བསྡུ་བའི་རྒྱས་སྤྱན་དྲངས་ལ་མཆོད་པ་ནི། ཨོཾ། ཉེར་སྤྱོད་བདུན་དང་ཤ་ལྔ་བདུད་རྩི་ལྔ། །ཟག་མེད་བདེ་ཆེན་མཆོད་འབུལ་བྱིན་གྱིས་རློབས། །ཨོཾ་བཛྲ་ཨརྒྷཾ། དྷཱུ་པེ། པུཥྤེ་ སོགས་ཀྱིས་མཆོད། ཛཿཧཱུྃ་བྃ་ཧོཿགཉིས་སུ་མེད་པར་གྱུར། མདུན་མཁར་ཡེ་ཤེས་དཀྱིལ་འཁོར་ལྷ་ཚོགས་རྣམས། །རྒྱུད་བསྐུལ་ཐུགས་ལས་མཁའ་འགྲོ་མ་ཚོགས་སྤྲོས། །མཚམས་བསྲུང་ཤིས་བརྗོད་བུམ་པས་དབང་བསྐུར་གྱུར། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨ་བྷི་ཥེ་ཀ་ཏ་ས་མ་ཡ་ཤྲཱི་ཡེ་ཧཱུྃ། ཆུ་ཞབས་ཡར་ལུད་གཙོ་བོར་སྣང་མཐའ་ཡས། །ཡུམ་ལ་རྣམ་སྣང་འཁོར་རྣམས་མི་བསྐྱོད་པ། །རིན་འབྱུང་འོད་དཔག་དོན་གྲུབ་རྒྱས་གདབ་གྱུར། །བདག་ཉིད་ཐུགས་ཀར་པདྨ་ལ་གནས་བའི། །ཧྲཱིཿདང་ཡུམ་གྱི་ཐུགས་ཀར་ཆོས་འབྱུང་དབུས། །ཧྲཱིཿལས་འོད་ཟེར་དཀར་སེར་དམར་ལྗང་འཕྲོས། །འཕགས་མཆོད་ཀུན་ཀྱང་ཐུགས་རྗེ་ཆེན་པོའི་སྐུ། །ཆར་བབས་རང་ཐིམ་སླར་ཡང་ཧྲཱིཿཡིག་འོད། །ཞལ་ནས་ཐོན་ཅིང་རིགས་བཞི་ཡབ་ཡུམ་དང༌། །མཁའ་འགྲོ་བཞི་ཡི་ཐུགས་ཀར་ཐིམ་སོ་སོའི། །ཐུགས་ལས་འོད་འཕྲོས་ཞལ་བརྒྱུད་བདག་ཡབ་ཡུམ། །ཐུགས་ཀའི་ཧྲཱིཿར་ཐིམ་ལས་བཞི་འགྲུབ་པར་གྱུར། །ཨོཾ་ཨཱཿཧཱུྃ་ཧྲཱིཿ ཨོཾ་མ་ཎི་པདྨེ་ཧཱུྃ། ཞེས་ཡི་གེ་བཅུ་པ་ཅི་ནུས་བཟླས། དྷུ་མ་གྷ་ཡེ་ན་མ་སྭཱ་ཧཱ། ཧ་རི་ནི་ས་ར་ཙ་ཧྲི་ཡ་ཧཱུྃ་ཛཿ ཞེས་དཔའ་བོ་མཁའ་འགྲོ་ཡོངས་ཀྱི་སྙིང་པོ་ཅི་རིགས་སུ་བཟླའོ། །དེ་ནས་ཐུན་མཐར་རྫོགས་རིམ་ནི། སླར་ཡང་རང་གི་ཐུགས་ལས་འོད་ཟེར་འཕྲོས། །འཁོར་འདས་ལྷར་གསལ་ལ་ཐིམ་དེ་དག་ཀྱང༌། །འོད་ཞུ་སྲུང་བའི་གུར་ཐིམ་རང་ལ་བསྡུ། །རང་ཡང་ཐུགས་ཀའི་ཧྲཱིཿཡིག་ལ་ཐིམ་པས། །ཧྲཱིཿཡང་འོད་ཞུ་གོང་བུ་ནཱ་ད་ལས། །མི་དམིགས་འོད་གསལ་སྤྲོས་བྲལ་ངང་དུ་བཞག ། ཅེས་ཅི་གནས་སུ་བཞག ། སླར་ཡང་རང་ཉིད་ཞལ་གཅིག་ཕྱག་གཉིས་པ། །སྤྱན་རས་གཟིགས་གསལ་ཕྱི་སྣོད་གཞལ་ཡས་ཁང༌། །ནང་བཅུད་ལྷ་དང་སྒྲ་གྲགས་སྔགས་ཀྱི་གདངས། །དྲན་རྟོག་ཏིང་འཛིན་ངང་ལ་ངོ་བོ་སྐྱོང༌། །སྤྱན་རས་གཟིགས་དབང་འཇིག་རྟེན་མགོན། །ཚད་མེད་ཐུགས་རྗེའི་རབ་བགྲོད་ཀྱིས། །དེང་འདིར་འགྲོ་དྲུག་སེམས་ཅན་རྣམས། །ཀུན་ཀྱང་མངོན་སངས་རྒྱས་གྱུར་ཅིག

། ཅེས་དང༌། བསྔོ་བ་སྨོན་ལམ་ཅི་བདེར་བྱའོ། །ཞེས་པའང་རྒྱལ་རོང་ཁྲོ་ཆུ་ཤུ་བྷཾ་ཆོས་གྲྭའི་སྒྲུབ་ཆེན་བླ་མ་བཛྲ་ལཀྵའི་གཡར་ངོར། སི་ཏུ་པདྨ་ཉིན་བྱེད་དབང་པོས་བཀོད་པ་དགེ་བར་གྱུར་ཅིག །སརྦ་མངྒ་ལཾ།། །། ༈ རྡོ་རྗེ་རྣམ་འབྱོར་མའི་བདག་བསྐྱེད་རྒྱུན་ཁྱེར་མཁའ་སྤྱོད་འཛུམ་ཞལ་བཞུགས། སྭ་བྷཱ་ཝ ས་སྦྱངས། སྟོང་ལས་ཁམས་གསུམ་ཆོས་འབྱུང་གཞལ་ཡས་དབུས། །པདྨ་རོ་ཉིའི་སྟེང་དུ་རང་སེམས་ཧྲཱིཿ །དམར་པོའི་འོད་ཀྱིས་དོན་གཉིས་བྱས་ཡོངས་གྱུར། །བདག་ཉིད་རྡོ་རྗེ་རྣལ་འབྱོར་མ་དམར་མོ། །ཞལ་གཅིག་ཁྲོ་གཉེར་སྤྱན་གསུམ་དམར་ལ་ཟླུམ། །དབུ་སྐྲ་ཁམ་ནག་འབར་འཕྱང་ཐོད་སྐམ་བརྒྱན། །མཆེ་གཙིགས་ཚེམས་ཀྱིས་མ་མཆུ་མནན་ཕྱག་གཉིས། །གྲི་གུག་ཐོད་ཁྲག་གྲུ་མོར་ཁ་ཊྭཱྃ་བརྟེན། །ཐོད་རློན་དོ་ཤལ་མེ་ཏོག་དྲིལ་གཡེར་དང༌། །ཕྱག་རྒྱ་ལྔས་བརྒྱན་གཡོན་བརྐྱང་གར་གྱིས་རོལ། །སྟོང་གསུམ་ཁྱབ་ཅིང་གཟི་འབར་གནས་གསུམ་དུ། །འབྲུ་གསུམ་མཚན་ལས་འོད་འཕྲོས་ཨོ་རྒྱན་ནས། །ཡེ་ཤེས་སྤྱན་དྲངས་དབང་བསྐུར་རྒྱས་གདབ་གྱུར། །གསང་བར་ཆོས་འབྱུང་པད་འདབ་བཞིར་ཧ་རི། །ནི་ས་དབུས་སུ་ཧྲཱིཿམཐར་སྔགས་ཕྲེང་གཡོན། །འཁོར་བས་མཆོག་ཐུན་དངོས་གྲུབ་ཐོབ་པར་གྱུར། །ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནི་ཡེ་ཧ་རི་ནི་ས་ཧཱུྃ་ཧཱུྃ་ཕཊ་སྭཱཧཱ། ཅི་ནུས་མཐར། སྣོད་བཅུད་འོད་ཞུ་བདག་ཐིམ་དེ་ཡང་ཧྲཱིཿ །ཐིག་ལེ་ནཱ་ད་མི་དམིགས་འོད་གསལ་ངང༌། །སླར་ཡང་བཅོམ་ལྡན་འདས་མའི་སྐུ་རུ་གསལ། །སྣང་གྲགས་ལྷ་ཡི་དཀྱིལ་འཁོར་ཉིད་དུ་ཤར། །དགེ་བ་འདི་ཡིས་མྱུར་དུ་བདག །རྡོ་རྗེ་ཕག་མོ་ སོགས་ནས་དགེ་བསྔོ་བྱ། དེ་ལྟར་བུ་རྒྱལ་གྱིས་བསྐུལ་ལྟར་སྤེལ་བ་དགེ། །། ༈ རང་ཉིད་ལྷར་གསལ་ཐུགས་ལས་རྃ་ཡྃ་ཁྃ། །ཚོགས་རྫས་ལ་ཐིམ་གཉིས་འཛིན་དྲི་མ་སྦྱངས། །ཨོཾ་ཨཱཿཧཱུྃ་གིས་ཡེ་ཤེས་བདུད་རྩིར་སྤེལ། །འདོད་ཡོན་ཟག་མེད་བདེ་ཆེན་མཆོད་སྤྲིན་གྱུར། ། ཅེས་པས་བྱིན་བརླབས་ལ།

ཧཱུྃ། དཀོན་མཆོག་ཀུན་འདུས་པདྨ་འབྱུང་གནས་མཆོག །རྩ་གསུམ་རྒྱལ་བའི་དཀྱིལ་འཁོར་གནས་འདིར་གཤེགས། །དམ་ཚིག་རྫས་མཆོག་པཉྩ་ཨ་མྲྀ་ཏ། །ག་ཎ་ཙཀྲའི་མཆོད་པ་ཆེན་པོ་འབུལ། །ཐུགས་དམ་བསྐང་ངོ་ནོངས་བཤགས་བར་ཆད་བགེགས། །དབྱིངས་སུ་དག་ཅིང་མཆོག་ཐུན་དངོས་གྲུབ་སྩོལ། ། ཅེས་ཚོགས་ཀྱི་མཆོད་པ་འབུལ་ནས་ལྷག་གཏོར་བསྔོ་བ་ནི། བཀའ་ཡི་ལྷག་གཏོར་དབང་རྣམས་འདིར་བྱོན་ལ། །ཚོགས་ལྷག་བཤལ་ཆུའི་དམ་བསྐང་ཕྲིན་ལས་མཛོད། ། ཅེས་པའང་ཚོགས་མཆོད་ཁྱེར་བདེར་ཨཱནནྡ་ཤཱ་ས་ནའི་བཞེད་ལྟར། སི་ཏུ་པདྨ་དེ་ཝེནྡྲའི་ལི་ཁི་ཏཾ། སརྦ་ཐཱ་མངྒ་ཀུརྦནྟུ།། །། ༈ རྒྱལ་ཡུམ་འཆི་མེད་མཆོག་སྟེར་འཕགས་མ་ཡིད་བཞིན་འཁོར་ལོའི་སྒོམ་བཟླས་ནི། སྐྱབས་ཡུལ་དཀོན་མཆོག་རྣམས་ལ་ཕྱག་འཚལ་ཞིང༌། །འགྲོ་བའི་དུཿཁ་སྒྲོལ་ཕྱིར་སེམས་བསྐྱེད་དོ། །ཨོཾ་སྭ་བྷཱ་ཝ་ སོགས་ཀྱིས་སྦྱངས་ལ། སྟོང་པའི་རང་མདངས་མ་འགགས་སྣང་བའི་རྩལ། །རང་བྱུང་རིག་པའི་ངོ་བོ་ཏཱྃ་ཡིག་དཀར། །དེ་ལས་འོད་འཕྲོས་འཕགས་མཆོད་འགྲོ་དོན་བྱས། །ཚུར་འདུས་ཡོངས་གྱུར་འཆི་མེད་མཆོག་སྦྱིན་ཡུམ། །སྟོན་མཚན་དབང་པོ་ཉ་རྒྱས་ལྟ་བུའི་མདོག །བརྒྱད་གཉིས་ལང་ཚོས་སྒེག་པ་ཞི་འཛུམ་ལྡན། །ཞལ་གཅིག་ཕྱག་གཉིས་དབུ་དང་ཕྱག་ཞབས་མཐིལ། །ཡེ་ཤེས་སྤྱན་བདུན་བཀྲ་ཞིང་དཀར་འཛིན་ཟུང༌། །རབ་རྒྱས་མཐོན་མཐིང་ལྕང་ལོ་ལྟག་པར་འཕྱང༌། །ཕྱག་གཡས་མཆོག་སྦྱིན་གཡོན་པས་ཨུཏྤ་ལ། །ལོ་འདབ་གེ་སར་བཞད་པ་ཐུགས་ཀར་བསྣམས། །མུ་ཏིག་གཙོར་གྱུར་རིན་ཆེན་སྣ་ཚོགས་ཀྱི། །དབུ་རྒྱན་དར་དཔྱངས་སྙན་ཆ་མགུལ་ཆུ་དང༌། །དོ་ཤལ་སེ་མོ་དོ་དང་ཕྱག་ཞབས་གདུབ། །སྐ་རགས་གཡེར་ཁའི་ཕྲེང་བས་མངོན་པར་མཛེས། །ལྷ་རྫས་དཀར་པོས་སྟོད་གཡོགས་སྣ་ཚོགས་དར། །སྨད་དཀྲིས་དང་བཅས་ཞབས་གཉིས་སྐྱིལ་མོ་ཀྲུང༌། །པད་ཟླའི་གདན་བཞུགས་ཙནྡྲའི་རྒྱབ་ཡོལ་གསལ། །གནས་གསུམ་འབྲུ་གསུམ་ཐུགས་ཀར་ཏཱྃ་ཡིག་མཚན། །དེ་ལས་འོད་འཕྲོས་པོ་ཊ་ལ་ཞིང་ནས། །ཡེ་ཤེས་སེམས་དཔའ་སྤྱན་དྲངས་དབྱེར་མེད་ཐིམ། །དབང་ལྷས་དབང་བསྐུར་སྣང་མཐས་རྒྱས་གདབ་གྱུར། །ཨོཾ་ཨཱརྻ་ཏཱ་རེ་བཛྲ་ཨརྒྷཾ་ཨཱཿཧཱུྃ།

སོགས་ནས། ཤབྡ འི་བར་གྱིས་མཆོད། སྙིང་གར་པད་ཟླའི་སྟེང་དུ་འཁོར་ལོ་དཀར། །རྩིབས་བརྒྱད་མུ་ཁྱུད་དང་བཅས་ལྟེ་བར་ཏཱྃ། །གཡས་སུ་ཨོཾ་དང་སྤེལ་ཚིག་སྔགས་མཐར་ཧཱ། །རྩིབས་རྣམས་སོ་སོར་ཡིག་འབྲུ་བརྒྱད་དེ་ལས། །འོད་འཕྲོས་རྒྱལ་བའི་བྱིན་རླབས་བརྟན་གཡོའི་བཅུད། །བསྡུས་ནས་འཁོར་ལོ་ཡིག་འབྲུར་ཐིམ་པར་གྱུར། །

ཡི་གེ་བཅུ་པ་ཅི་ནུས་བཟླ་ཞིང༌། སྔགས་སྤེལ་ཚིག་ཅན་ཡང་བཟླའོ། །ཐུན་མཐར། སྣང་སྲིད་ཡེ་ཤེས་ལྷ་ཡི་དཀྱིལ་འཁོར་པ། །བདག་ཉིད་ལ་ཐིམ་རང་ལྷར་གསལ་བ་ཡང༌། །ཡས་མར་འདུས་ནས་ཐུགས་ཀའི་ཏཱྃ་ལ་ཐིམ། །དེ་ཡང་འོད་ཞུ་ཐིག་ལེ་ནཱ་དར་བསྡུ། །སེམས་ཉིད་མ་བཅོས་སྤྲོས་དང་བྲལ་བའི་དབྱིངས། །དམིགས་མེད་བདེ་གསལ་མི་རྟོག་ངང་དུ་བཞག ། ཅེས་མཉམ་པར་བཞག་ལ་རྗེས་ཐོབ་ནི། སླར་ཡང་སྣང་སྟོང་ཟུང་འཇུག་ལྷ་སྐུར་ལྡང༌། །འཇིག་རྟེན་དབང་སྲས་འཕགས་མའི་གོ་འཕང་ལ། །མཁའ་དང་མཉམ་པའི་སྲིད་འགྲོ་ཇི་སྙེད་པ། །འགོད་ཕྱིར་འཁོར་གསུམ་དམིགས་མཐའ་བྲལ་བར་བསྔོ། ། ཅེས་པའང་སི་ཏུ་པས་རྒྱལ་བའི་ཕྲིན་ལས་དབོན་གང་དེའི་བཞེད་ལྟར་བཀོད་པ་ཤུ་བྷ་པུཥྚིཾ།། །། ༈ སྣང་སྲིད་གཉིས་བསྡུས་ཆོས་རྣམས་ཀུན། །སྟོང་པར་གྱུར་པ་དེའི་ངང་ལས། །བྃ་ལས་ཡན་ལག་བརྒྱད་ལྡན་མཚོར། །པྃ་ལས་པད་དཀར་འདབ་སྟོང་ལྡན། །ཨ་ལས་ཟླ་དཀྱིལ་སྟེང་དུ་ནི། །རང་སེམས་ཏཱྃ་ལས་འོད་འཕྲོས་པས། །དོན་གཉིས་བྱས་ཏེ་འཕགས་མའི་སྐུ། །ཆུ་ཤེལ་སྟོན་ཟླའི་ཁ་དོག་ཅན། །ཞལ་གཅིག་ཕྱག་གཉིས་གཡས་མཆོག་སྦྱིན། །གཡོན་པའི་མཐེབ་སྲིན་ཨུཏྤལ་བསྣམས། །སྤྱན་བདུན་བཀྲ་ཞིང་ལྷ་རྫས་ཀྱི། །དར་དཀར་སྟོད་གཡོགས་སྣ་ཚོགས་དར། །སྨད་གཡོགས་དབུ་རྒྱན་སྙན་རྒྱན་དང༌། །མགུལ་རྒྱན་ཕྱག་ཞབས་གདུ་བུ་སོགས། །རིན་ཆེན་རྒྱན་ཀུན་རབ་མཛེས་པ། །ཞབས་གཉིས་རྡོར་སྐྱིལ་ཟླ་བ་ཡི། །རྒྱབ་ཡོལ་འོད་དང་འོད་ཟེར་འཕྲོ། །གནས་གསུམ་ཨོཾ་ཨཱཿཧཱུྃ་ཡིག་དང༌། །ཐུགས་ཀའི་ཆ་སྨད་ཏཱྃ་ཡིག་མཚན། །འོད་འཕྲོས་པོ་ཊ་ལ་ཞིང་ནས། །རང་འདྲའི་ལྷ་མོ་དཔག་མེད་པ། །སྤྱན་དྲངས་དམ་ཡེ་གཉིས་མེད་བསྟིམ། །རང་གི་ཐུགས་ལས་འོད་འཕྲོས་པས། །དབང་ལྷས་དབང་བསྐུར་མཆོད་པས་མཆོད། །རིགས་བདག་འོད་དཔག་དབུ་རྒྱན་གྱུར། །ཐུགས་ཀར་ཟླ་སྟེང་འཁོར་ལོ་དཀར། །རྩིབས་བརྒྱད་མུ་ཁྱུད་དང་བཅས་པ། །དབུས་སུ་ཏཱྃ་མཐར་སྔགས་ཕྲེང་དང༌། །རྩིབས་བརྒྱད་ཡིག་བརྒྱད་འགྲེང་ཐབས་འཁོད། །འོད་ཀྱིས་རྒྱལ་མཆོད་སེམས་ཅན་ཀུན། །སྒྲིབ་སྦྱང་བརྟན་གཡོ་སྣོད་བཅུད་ཀྱི། །དཔལ་འབྱོར་བསྡུས་ནས་ཡིག་འབྲུར་ཐིམ། །ཚེ་དབང་ཡེ་ཤེས་དབང་ཐོབ་གྱུར།

།ཡི་གེ་བཅུ་པ་ཅི་ནུས་བཟླས་ལ། ཕྱི་ནང་སྣོད་བཅུད་དངོས་པོ་རྣམས། །འཕགས་མའི་དཀྱིལ་འཁོར་ཡོངས་ཤར་བ། །ཀུན་ཀྱང་རང་ཐིམ་འོད་ཞུ་ནས། །ཏཱྃ་ཡིག་ནཱ་ད་འོད་གསལ་བ། །དམིགས་མེད་སྤྲོས་བྲལ་བདེ་བ་ཆེ། །གང་ཤར་ཆོས་སྐུའི་ངང་དུ་བཞག །སླར་ཡང་རྗེ་བཙུན་འཕགས་མའི་སྐུ། །ཡོངས་གྱུར་གནས་གསུམ་ཡིག་གསུམ་མཚན། །གཟུང་འཛིན་དངོས་ཀུན་ཆོས་རྣམས་ཀུན། །ལྷ་སྔགས་ཡེ་ཤེས་རོལ་པར་ཤར། །དགེ་བ་འདི་ཡིས་ སོགས། ཅེས་གནས་མདོ་ཀྱུ་ར་དགེ་སློང་ཀརྨ་ཡེ་ཤེས་ངོར་སྤེལ།། །། ༈ སྟོང་པའི་ངང་ལས་པད་ཉི་ལ་ཧྲཱིཿལས་སྒྲོལ་མ་དམར་མོ། ཕྱག་གཡས་སྐྱབས་སྦྱིན་དང༌། གཡོན་ལྕགས་ཀྱུས་མཚན་པའི་ཨུཏྤལ་འཛིན་པ། ལྷ་དང་མི་ལས་འདས་པའི་རྒྱན་དུ་མས་སྤྲས་པ། ཞབས་རོལ་པའི་སྟབས་ཀྱིས་མི་ཡི་དབང་པོ་མནན་པར་ བསམ་ལ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སྭཱ་ཧཱ། ཞེས་ཅི་ནུས་བཟླས་མཐར། སྨོན་ལམ་ནི། བཅོམ་ལྡན་སྒྲོལ་མ་དབང་མཛད་དཔལ་མོ་ལ། །དེང་བཟུང་སྙིང་པོའི་བར་དུ་སྐྱབས་གསོལ་འདེབས། །རྒྱལ་པོས་འཇིགས་སོགས་བར་ཆད་ནད་གདོན་ཞི། །ཚེ་བསོད་འབྱོར་རྒྱས་བྱང་ཆུབ་ཆེ་ཐོབ་ཤོག ། ཅེས་སི་ཏུ་པདྨའི་མིང་གིས་དད་གསུམ་ལྡན་པ་རྨད་དགའ་ལ་སྤེལ་བ། མ་གཞི་ཨ་རཱ་གའི་གསུངས་པ་གཞིར་བཞག །སྨོན་གསོལ་སྤེལ་བ་དགེའོ།། །། ༈ སྭསྟི། མཆོག་གསུམ་རྒྱལ་ལ་གུས་འདུད་འགྲོ་བ་རྣམས། །འབྱུང་བ་ཞུགས་ལྕེའི་འཇིགས་ལས་སྐྱོབ་པའི་ཕྱིར། །ཏཱ་ར་ཀྲོ་དྷིའི་བསྐྱེད་བཟླས་ཟབ་གནད་དོན། །འཇུག་བདེ་འདིར་བཀོད་བཀའ་ཡི་གནང་བ་སྩོལ། ། སྭ་བྷ་ཝ ས་སྦྱངས། ཆོས་རྣམས་དམིགས་མེད་སྟོང་པའི་རང་མདངས་ལས། །སྣ་ཚོགས་པདྨ་ཟླ་བའི་དཀྱིལ་འཁོར་སྟེང༌། །རང་ཉིད་སྐད་ཅིག་འཕགས་མ་སྒྲོལ་མ་མཆོག །རབ་ཁྲོས་པདྨ་རཱ་གའི་རི་བོ་ལྟར། །དྭངས་ཤིང་གསལ་ལ་འོད་འཚེར་ཞལ་བདུན་ལྡན། །མཆེ་གཙིགས་ལྗགས་བསྐྱོད་རལ་གཡེངས་སྨིན་མ་འབར། །ཕྱག་གཉིས་ནོར་བུ་ཆུ་ཤེལ་བདུད་རྩི་ཅན། །འོད་དཀར་བསིལ་བའི་རང་བཞིན་དཀྱིལ་འཁོར་བསྣམས། །དར་དང་རིན་ཆེན་རུས་པའི་རྒྱན་དང་ནི། །སྟག་གི་ཤམ་གསོལ་ཞབས་གཉིས་རོལ་སྟབས་ཀྱི། །འོད་དང་འོད་ཟེར་སྣ་ཚོགས་ཀློང་དུ་གསལ། །གནས་གསུམ་འབྲུ་གསུམ་ཐུགས་ཀའི་ཧཱུྃ་འོད་ཀྱིས། །བསྐུལ་བས་ཡེ་ཤེས་སྤྱན་དྲངས་དབྱེར་མེད་གྱུར། །ཐུགས་ཀར་ཟླ་སྟེང་ཡིག་བཅུའི་སྔགས་ཕྲེང་ལས། །འོད་འཕྲོས་རང་གཞན་བསྐལ་པ་ནས་བསགས་པའི། །སྡིག་སྒྲིབ་ཉེས་ཚོགས་འཕྲལ་གྱི་མི་མཐུན་ཕྱོགས། །འཇིགས་བརྒྱད་བཅུ་དྲུག་ཞི་ཞིང་ཨགྣེ་ཡེ། །བར་ཆད་ལས་རྒྱལ་སྣོད་བཅུད་དག་པའི་ཞིང༌། །པོ་ཊ་ལ་ཡི་གནས་མཆོག་མངོན་གྱུར་ནས། །སྣང་གྲག་རྟོག་ཚོགས་ལྷ་ཡི་དཀྱིལ་འཁོར་ཤར། །གཟུངས་སྔགས་རང་སྒྲས་སྣང་སྲིད་ཞི་བ་དང༌། །བྱམས་པའི་ངང་ལས་དགེ་ལེགས་སྐྱེད་པར་གྱུར། །ཨོཾ་ཤྲི་ཁཱི་བ་ད་ནེ་མ་ཧཱ་ཨུལྐ་མུ་ཁི་ཛྭ་ལ་མ་ལི་ནཱ་སཔྟ་ཛི་ཧྭཱི།

སཔྟ་མུ་ཁཱི་བི་ཤྭ་ན་རི་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སྭཱ་ཧཱ། ཅི་ནུས་བཟླ་ཞིང་ལྷ་ཡི་ང་རྒྱལ་བརྟན་པོས་དམིགས་པའི་གསལ་འདེབས་གང་འོས་བྱའོ། ། མཐར་ནི་ལྷ་སྐུ་མི་དམིགས་སྟོང་པའི་ངང༌། །མ་བཅོས་སྤྲོས་པ་བྲལ་བའི་ཀློང་དུ་བཞག །དགེ་བས་བདག་གཞན་སྔོན་བསགས་འཕྲལ་ལས་དབང༌། །རྐྱེན་ངན་གྲོལ་ཞིང་བསམ་དོན་ཆོས་བཞིན་འགྲུབ། །མཁའ་མཉམ་འགྲོ་བ་རྣམ་པར་གྲོལ་བའི་ལམ། །བཀོད་པའི་ཕྲིན་ལས་གེགས་མེད་ཉིད་འགྲུབ་ཤོག ། ཅེས་པའང་རིག་འཛིན་བླ་མའི་ཐུགས་ཀྱི་སྲས་མཆོག་ཚེ་དབང་དཔལ་དར་མཆོག་སྤྲུལ་གྱི་བཀའི་བཞེད་ལྟར། སི་ཏུ་པདྨ་ཉིན་བྱེད་དབང་པོས་བཀོད་དེ་ཕུལ་པ་མངྒ་ལཾ།། །།
