༄༅། །བསྟན་པའི་བཀའ་སྲུང་ཨ་ཕྱི་ཆེན་མོའི་མཆོད་ཕྲེང་བཀྲ་ཤིས་ཆར་འབེབས་ཞེས་བྱ་བ་བཞུགས་སོ། །

ན་མོ་གུ་རུ་པདྨ་དེ་ཝ་ཌཱ་ཀི་ནི་ཡེ། འདིར་ཡེ་ཤེས་ཀྱི་མཁའ་འགྲོ་མ་དྷརྨ་ཏཱ་རའི་མཆོད་ཕྲེང་ལས་བྱང་གི་རིམ་པ་སྤེལ་བ་ལ། ཐོག་མར་དུར་ཁྲོད་ལ་སོགས་པ་ཉེར་བཞིའི་གནས་སམ། གྲུབ་ཐོབ་བླ་མས་བྱིན་གྱིས་བརླབས་པའི་གནས་སམ་ཡིད་དང་མཐུན་པའི་གནས་དབེན་པར། ཚེས་བཅུ་ལ་སོགས་པའི་དུས་ཚེས་མཁའ་འགྲོ་འདུ་བའི་གཟའ་སྐར་དུས་སྦྱོར་བཟང་པོ་ལ་གཙང་སྦྲ་ལེགས་པར་བྱས་ལ། རྩ་བ་གསུམ་གྱི་རྟེན་དང༌། ཁྱད་པར་ཨ་ཕྱི་དང༌། ཚེ་རིང་མའི་ཐང་སྐུ་དང་སྐུ་འབག་ལ་སོགས་པ་དང༌། དར་སྣ་སྟག་གཟིག་དཀོར་ནོར་སྤྱན་གཟིགས་ཀྱང་ཅི་འབྱོར་པས་བརྒྱན་ཅིང༌། ལྷ་གྲངས་དང་མཉམ་པའི་ཚོམ་བུའམ། ཡང་ན་ལྔ་ཚོམ། འབྲུ་དཀར་གྱི་ཕྱེ་མ་རྩི་བཅུད་ ཀྱིས་སྦྲུས་ཏེ་གཏོར་མ་ཟླུམ་པོ་ལྔ། འཇའ་འོད་བཀྲ་བའི་རྒྱན་དང༌། རྩ་བ་གསུམ་གྱི་གཏོར་མ་དང༌། སྨན་གཏོར་དང༌། བསྐུལ་གྱི་དམར་གཏོར་གཉིས་དང༌། ཆད་མདོ། བརྟན་སྐྱོང༌། ཕྱོགས་སྐྱོང་གཏོར་མ། སྔོན་འགྲོ་གཏོར་མ་སོགས་དང༌། སྤྲོ་ན། བསྐང་གཏོར་བང་རིམ་གཞིའི་སྟེང༌། དབུས་མར་ཨ་ཕྱི་གཙོ་འཁོར་ལྔའི་ཟླུམ་གཏོར་དང༌། མཚམས་བཞིར་ཐུགས་སྤྲུལ་གསུམ་དང་སྤུ་གྲི་མ་བཞིའི་གཏོར་མ་ཟུར་གསུམ་བཞི། དེ་ཕྱིར་ཚེ་རིང་མཆེད་ལྔའི་ཟླུམ་གཏོར། མདུན་དུ་ཁྲོ་མོའི་གཏོར་མ་ཟུར་གསུམ་ལིངྒས་མཚན་པ། གཉིས་པར་བརྟན་མ་བཅུ་གཉིས་ལ་བཤོས་བུ་ཟུར་གསུམ་བཅུ་གཉིས། དེ་མཐར་དབང་ཕྱུག་ཉེར་བརྒྱད་ལ་རྐང་ཟོར་གྱི་རྣམ་པ་རང་མདོག་གིས་བསྐོར་བའི་ལོག་བཞིར་ཚོགས་ཀྱི་རྣམ་པ་བཞི་དང༌། མཐར་མཐེབ་ཀྱུས་བསྐོར། གསུམ་པ་མདུན་ནས་གཡོན་སྐོར་དུ། མདའ་དར་མེ་ལོང་བརྒྱན་པ། ཐོད་ཌཱར། ནོར་བུ་འབར་བ། ཐོད་ཁྲག །སྨན་ཐོད། རྡོ་རྗེ། ཁ་ཊཱྃ་ག །ན་བཟའ། རུས་ཐོད་དང་རྒྱན་ཕྲེང༌། རྐང་གླིང༌། སྦ་ལྕག །རིན་པོ་ཆེ། གསེར་ཌཱར། ལྕགས་ཀྱུ། གྲི་གུག །མགོ་ཕྲེང༌། ནེའུ་ལེ། ཐོད་པ་ནོར་བུས་གང་བ། བེར། དུར་ཁྲོད་ཆས་བརྒྱད། རལ་གྲི། སྙིང༌། ནད་རྐྱལ། བཞི་པ་རྟ་སྔོ་སེར་དམར་ནག་སྒ་ཆས་ལྡན།

འབྲོང༌། མཁའ་ལྡིང༌། བོང་བུ། སྤྲེའུ། སྤྱང་ཀི། སེང་གེ །སྟག་འབྲུག །གླ་བ། གསེར་ཤ་ཡུ་མོ། རྐྱང༌། ཁྲུང་ཁྲུང༌། གཞན་ཡང་དཀོར་ནོར་རིགས་གང་འབྱོར་བཀོད། ཕྱོགས་བཞིར་ཁྲུང་ཁྲུང་རྒྱལ་མཚན། མདའ་དར་མེ་ལོང་གིས་བརྒྱན་པ་རྣམས་བཙུགས། མཐར་མཆོད་པ་སྣ་ཚོགས་དང་མཆོད་གཏོར་རྣམས་བཤམས། ནང་མཆོད་རྩ་བརྒྱད་སྦྱར་བའི་བདུད་རྩི་དང༌། འབྲུ་སྨན་རིན་པོ་ཆེས་སྦྲུས་པའི་གཏོར་མ་དང་རཀྟ་སོགས་ཀྱང་བཤམས། ཚོགས་གཏོར་དང་དཀར་མོའི་ལག་གཡས། འབྲུ་ཡི་ཆང་གི་ཉིང་ཁུ། གཞན་ཡང་འབྲུ་སྣ། སྨན་སྣ། དར་ཚོན། ཟས་སྣ་ཤིང་ཐོག་གསེར་སྐྱེམས་ལ་སོགས་པ་བཤམས་ནས་རང་ཉིད་བདེ་བའི་སྟན་ལ་འཁོད་ནས་ཐོག་མར་སྔོན་འགྲོ་སྐྱབས་སེམས་ཀྱི་རིམ་པ་ནི། ན་མོ། མཆོག་གསུམ་རྩ་གསུམ་རབ་འབྱམས་ལ། །རྩེ་གཅིག་སྒོ་གསུམ་དད་པའི་བློས། །དང་བའི་ཡིད་ཀྱིས་སྐྱབས་སུ་མཆི། །འགྲོ་ཀུན་ཕྱིར་མི་ལྡོག་པ་ཡི། །བྱང་ཆུབ་གོ་འཕང་ཐོབ་པར་ཤོག ། ལན་གསུམ། དེ་ནས་གནས་བདག་དང་ལན་ཆགས་ལ་གཏོར་མ་བསྔོ་བ་ནི། ཨོཾ་ཨ་ཀཱ་རོ ་ལན་གསུམ་བྱིན་གྱིས་བརླབས་ལ། ཧཱུྃ། ཕྱོགས་སྐྱོང་ཡུལ་ལྷ་སྡེ་བརྒྱད་རྣམས། །མཆོད་གཏོར་བཟང་པོ་འདི་ལོངས་ལ། །དཀར་ཕྱོགས་ཀུན་གྱིས་མཐུན་རྐྱེན་སྤེལ། །ནག་པོའི་ཕྱོགས་གྱུར་འབྱུང་པོའི་རིགས། །བསེ་རག་བྲེལ་འཕོངས་ཐེའུ་རང་སོགས། །སོ་སོར་མོས་པའི་བཟའ་དང་བཏུང༌། །ལན་ཆགས་གླུད་གཏོར་འདི་ཁྱེར་ལ། །མཐའ་མེད་གནས་སུ་ཕྱིར་དེངས་ཤིག །གལ་ཏེ་མི་ཉན་འཚེ་བྱེད་ན། །ང་ཉིད་རྡོ་རྗེ་སྲིན་པོའི་སྐུ། །དབང་ཆེན་ཁྲོས་པའི་རྔམས་ཟེར་གྱིས། །བགེགས་ཚོགས་ཐལ་བར་བརླག་ཏ་རེ། །ཨོཾ་པདྨན་ཏ་ཀྲིཏྟ་ཧཱུྃ་ཕཊ། ཨོཾ་སུམྦྷ་ནི་ སོགས་ཧཱུྃ་བཞིའི་སྔགས་བརྗོད་ཡུངས་ཀར་བྲབ་ལ་གུ་གུལ་འདུག །སྲུང་འཁོར་ནི། རང་གི་ཐུགས་ཀའི་འོད་ཟེར་གྱི། །ཡེ་ཤེས་མེ་དཔུང་རྡོ་རྗེ་དང༌། །ལྕགས་སྡིག་ཕུར་བུའི་ཚོགས་སྤྲོས་པས། །སྲ་བརྟན་གཞོམ་བྲལ་སྲུང་འཁོར་གྱུར། །བཛྲ་རཀྵ་རཀྵ་སྭཱ་ཧཱ།

དེ་ནས་མཆོད་རྫས་རྣམས་བསང་སྦྱང་སྤེལ་བ་དང་བྱིན་གྱིས་བརླབ་པ་ནི། ཨོཾ་པདྨན་ཏ་ཀྲི་ཏ་ཧཱུྃ་ཕཊ་ ཀྱིས་བསངས་ལ། སྭ་བྷཱ་ཝ ས་སྦྱངས། སྟོང་པར་གྱུར་པའི་ངང་ཉིད་ལས། །བྷཱུྃ་ལས་རིན་ཆེན་པདྨའི་སྣོད། །ཡངས་ཤིང་རྒྱ་ཆེ་རྣམས་ཀྱི་ནང༌། །མཆོད་པའི་རྫས་གྱུར་ཡིད་དུ་འོང། །མཆོད་ཡོན་ཞབས་བསིལ་མེ་ཏོག་དང༌། །བདུག་སྤོས་སྣང་གསལ་དྲི་ཆབ་དང༌། །ཞལ་ཟས་རོལ་མོ་གཟུགས་སྒྲ་དྲི། །རོ་བཅུད་རེག་བྱ་འདོད་ཡོན་ལྔ། །བཀྲ་ཤིས་རྫས་རྟགས་རྒྱལ་སྲིད་བདུན། །སྨན་དང་རཀྟ་བ་ལིང་ཏ། །བསྐང་རྟེན་སྒྲུབ་རྫས་ཕུན་སུམ་ཚོགས། །འབྲུ་གསུམ་འོད་དུ་ཞུ་བ་ལས། །ཟག་མེད་ཡེ་ཤེས་བདུད་རྩི་དངོས། །ཕྱོགས་ཀུན་རྒྱས་པར་འགེངས་པ་ཡི། །ནམ་མཁའི་མཛོད་བཞིན་ལོངས་སྤྱོད་ཚོགས། །རྒྱ་ཆེན་རྒྱུན་མི་ཆད་པར་གྱུར། །ཨོཾ་བཛྲ་ཨརྒྷཾ་ཨཱཿཧཱུྃ་སྭཱ་ཧཱ། དེ་བཞིན་དུ། པདྱཾ། པུཥྤེ། དྷུ་པེ། ཨ་ལོ་ཀཻ། གནྡྷེ། ནེ་ཝི་དེ། ཤབྡ། རཱུ་པ། ཤབྡ། གནྡྷེ། ར་ས། སྤརྵེ་ཨཱཿཧཱུྃ་སྭཱ་ཧཱ། སརྦ་པཉྩ་ཨ་མྲྀ་ཏ་ཨཱ་ཧཱུྃ་སྭཱ་ཧཱ། མ་ཧཱ་རཀྟ་ཛྭ་ལ་ཨཱཿཧཱུྃ་སྭཱ་ཧཱ། མ་ཧཱ་བ་ལིང་ཏ་ཨཱཿཧཱུྃ་སྭཱ་ཧཱ། དེ་ནས་རང་ཉིད་ལྷ་བསྐྱེད་པའི་རིམ་པ་ནི། ཨོཾ་ཤཱུ་ནྱ་ཏ་ཛྙཱ་ཎ་བཛྲ་སྭཱ་བྷཱ་ཝ་ཨདྨ་ཀོ྅ཾ། སྣང་སྲིད་དངོས་ཀུན་ཆོས་ཐམས་ཅད། །སྟོང་པ་ཉིད་གྱུར་དེའི་ངང་ལས། །པད་དཀར་ཟླ་ཉིའི་གདན་སྟེང་དུ། །ཧྲཱིཿལས་འོད་འཕྲོས་སངས་རྒྱས་མཆོད། །སེམས་ཅན་ཀུན་གྱི་སྒྲིབ་པ་སྦྱངས། །ཚུར་འདུས་རང་ཉིད་སྐད་ཅིག་གིས། །ཨོ་རྒྱན་སློབ་དཔོན་རིག་པ་འཛིན། །པདྨ་ཐོད་ཕྲེང་སྐུ་མདོག་དཀར། །ཞལ་གཅིག་ཕྱག་གཉིས་གཡས་རྡོ་རྗེ། །གཡོན་པས་ཐོད་པ་བཅུད་བཀང་དང༌། །ཁ་ཊཱྃ་དཀུར་བརྟེན་ཟ་འོག་གི །ཆོས་གོས་རྣམ་གསུམ་ཕོད་ཀ་དང༌། །པད་ཞྭ་རིན་ཆེན་རྒྱན་སྤྲས་གསོལ། །སྙན་ཆ་མགུལ་རྒྱན་ཕྱག་ཞབས་གདུབ། །དར་དཔྱངས་ཅོད་པན་ལྷབ་ལྷུབ་གཡོ། །ཞབས་གཉིས་རྡོ་རྗེ་སྐྱིལ་ཀྲུང་བཞུགས། །གནས་གསུམ་ཨོཾ་ཨཱཿཧཱུྃ་གིས་མཚན། །འོད་འཕྲོས་ཆོས་དབྱིངས་དག་ཞིང་ནས། །བླ་མ་སྐུ་གསུམ་འདུས་པའི་དཔལ། །མཚོ་སྐྱེས་དྲི་མེད་རྡོ་རྗེ་འཆང༌། །རྒྱལ་བ་སྲས་དང་བཙུན་མོར་བཅས། །ལྔ་ལྡན་འཇའ་འོད་ཐིག་ལེ་དང༌། །ཐིག་ཕྲན་མེ་ཏོག་ཆར་ཆེན་དབུས། །དཔའ་བོ་མཁའ་འགྲོ་གིང་གི་ཚོགས། །སྤྲིན་ཆེན་རླུང་གིས་བསྐྱོད་བཞིན་དུ། །དམ་ཚིག་ཡེ་ཤེས་དབྱེར་མེད་པར། །ཆུ་ལ་ཆུ་བཞག་ལྟ་བུར་གྱུར། །སླར་ཡང་ཐུགས་ཀའི་འོད་ཟེར་གྱིས། །ལྷ་དང་ལྷ་མོའི་ཚོགས་སྤྲོས་ནས། །གསང་གསུམ་དགྱེས་པ་བསྐྱེད་པའི་ཕྱིར། །མཆོད་པའི་བྱེ་བྲག་དུ་མ་ཡིས། །མཆོད་ཅིང་སྙན་པའི་དབྱངས་ཀྱིས་བསྟོད། །ཐུགས་ཀར་རྡོ་རྗེ་རྩེ་ལྔའི་ལྟེར། །ཧྲཱིཿམཐར་སྔགས་ཕྲེང་གཡས་སུ་འཁོར། །འོད་ཀྱིས་ཕྱོགས་བཅུའི་སངས་རྒྱས་དང༌། །བྱང་སེམས་བླ་མ་ཡི་དམ་དང༌། ཞེས་བསྙེན་པ་ཅི་ནུས་སུ་བཟླས་ལ། མ་ཡེངས་པའི་ངང་ནས་ཁྱེར་སོ་གསུམ་དང་མ་བྲལ་བར་བྱའོ། །དེ་ནས་རང་ཉིད་ལྷག་པའི་ལྷ་ཡི་ང་རྒྱལ་ལ་གནས་བཞིན་པ་དེ་ཡི་མདུན་དུ་མཁའ་འགྲོ་རྣམས་གསལ་བཏབ་ཅིང་བསྐྱེད་པ་ནི་འདི་སྐད་དོ།། ཧྲཱིཿ རང་ཉིད་ལྷར་གསལ་མདུན་གྱི་ས་གཞིའི་དབུས། །འཇའ་ཟེར་འོད་ཀྱི་མདངས་འཚེར་ཀུན་ཏུ་དཀྲིགས། །ཤེལ་གྱི་གུར་ལ་གཡུ་ཡི་ཆོན་ཕུར་བཏབ། །མུ་མེན་འཕྲུལ་གྱི་པ་ཏྲ་རིས་བསྐོར་ཞིང༌། །ཀ་ན་ཀ་ཡི་རྒྱན་གྱིས་མཛེས་པ་ཡི། །ཡིད་འོང་བསྟི་གནས་འཆི་མེད་ཞིང་བདག་པོའི། །སྐྱེད་ཚལ་ལས་ཀྱང་ཆེས་ཆེར་རྨད་བྱུང་བའི། །ལེགས་པའི་དཔལ་གྱི་ཡོན་ཏན་ཕུལ་དུ་གྱུར། །དེ་དབུས་ཆུ་འཛིན་དྲ་བ་སྤེལ་བའི་ནང༌། །ཉི་ཟེར་པདྨ་རྒྱས་པའི་གདན་སྟེང་དུ། །ཏི་ཡིག་ཡོངས་གྱུར་མྱུར་མགྱོགས་ཤུགས་ལྡན་པའི། །ཅང་ཤེས་དུང་མདོག་གཡུ་རྔོག་དར་སྟན་ཅན། །གསེར་སྒ་གཡུ་ཡི་སྲབ་མཐུར་ཤེལ་ཡོབ་དང༌། །རིན་ཆེན་གསེར་དངུལ་ལ་སོགས་དུ་མ་ཡི། །རྒྱན་ལྡན་སྟེང་དུ་ཧྲཱིཿཡིག་ཡོངས་གྱུར་ལས། །དཔལ་གྱི་ལྷ་མོ་མཁའ་འགྲོ་ཀུན་གྱི་གཙོ། །མ་གཅིག་དབང་ཕྱུག་ཆོས་ཀྱི་སྒྲོལ་མ་ནི། །ཉ་རྒྱས་ཆུ་ཤེལ་འོད་ཀྱི་སྐུ་མདོག་ལ། །དབུ་སྐྲ་ལྕང་ལོ་ཐོར་ཚུགས་དུང་ཐོད་ཅན། །ཞི་མ་ཁྲོ་འཛུམ་ཞལ་གཅིག་སྤྱན་གསུམ་ལྡན། །ཕྱག་གཉིས་གཡས་ཀྱིས་ཐོད་པའི་ཅང་ཏེའུ་འཁྲོལ། །གཡོན་གྱིས་ཐོད་པ་མཚན་ལྡན་ནོར་བུ་ཡིས། །བཀང་པ་བསྣམས་ཞིང་དར་དཀར་དྲི་མེད་ཀྱི། །ན་བཟའ་བཀྲ་ཤིས་རེ་ཁཱ་བཀྲ་བ་ཅན། །ལྷབ་ལྷུབ་དར་སྣ་ལྔ་ཡི་ཅོད་པན་དང༌། །གསེར་དང་ཨིནྡྲ་ནཱི་ལ་མུ་ཏིག་གཡུ། །མཐོན་ཀ་མརྒད་ཕ་ལམ་ཀེ་ཀེ་རུ། །པདྨ་རཱ་ག་ལ་སོགས་རིན་ཆེན་དང༌། །མཆོག་གི་ནོར་བུ་བསམ་འཕེལ་འདོད་འབྱུང་གིས། །དབུ་རྒྱན་སྙན་ཆ་མགུལ་ཆུ་སྐ་རགས་དང༌། །དོ་ཤལ་སེ་མོ་དོ་དང་ཕྱག་ཞབས་ཀྱི། །གདུ་བུ་དངུལ་གྱི་གཡེར་ཁའི་ཕྲེང་བས་བརྒྱན། །གསང་གསུམ་ཡོན་ཏན་རབ་རྫོགས་སྐུ་ཅན་གྱུར། །སྤྲུལ་པའི་ཚུལ་སྟོན་ཡེ་ཤེས་ཌཱ་ཀི་མ། །འབྲོང་ཆེན་ལ་ཞོན་གཙུག་ན་ནོར་བུ་ནི། །དཀར་སེར་ཞལ་གཅིག་ཕྱག་གཉིས་རིན་ཆེན་དང༌། །ཐོད་པ་ནོར་བུས་བཀང་འཛིན་དར་བེར་ལྗང༌། །གསོལ་ཞིང་རྒྱན་རྣམས་ཀུན་གྱིས་རྣམ་པར་བཀླུབས། །ཅང་ཤེས་མཚལ་བུ་ལ་ཞོན་དམ་ཚིག་གི

།མཁའ་འགྲོ་ཡིད་བཞིན་དམར་སྐྱ་ཞལ་གཅིག་པ། །ཕྱག་གཉིས་ནེའུ་ལེ་ཐོད་པར་རིན་ཆེན་གཏམས། །དར་དམར་བེར་གསོལ་མཛེས་པའི་རྒྱན་གྱིས་སྤྲས། །མཁའ་ལྡིང་ལ་ཆིབས་དབང་གི་མ་མོ་ནི། །ཁམས་གསུམ་གཟི་འཕྲོག་དམར་མོ་ཞལ་གཅིག་པ། །ཕྱག་གཉིས་ཟངས་གྲི་ལྕགས་ཀྱུ་དར་བེར་གསོལ། །ཐོད་པ་སྐམ་རློན་རིན་ཆེན་རྒྱན་གྱིས་སྤུད། །བོང་བུ་གྭ་དམར་ལ་ཞོན་ཤ་ཟ་མོ། །ཀརྨ་ཌཱ་ཀི་མཐིང་ནག་རྔམ་ཞལ་གཅིག །ཕྱག་གཉིས་རལ་གྲི་བྷནྡྷ་ཙིཏྟ་རོལ། །གླང་ཀོ་ཞིང་ལྤགས་དུར་ཁྲོད་དཔའ་ཆས་བརྗིད། །ཀུན་ཀྱང་མཐུ་སྟོབས་གཟི་མདངས་གསལ་བར་བསྐྱེད། །ཧྲཱིཿ སྨན་བཙུན་ཌཱ་ཀི་ཆོས་སྲུང་ཡོངས་ཀྱི་གཙོ། །བཀྲ་ཤིས་ཚེ་རིང་དཔལ་སྟེར་ལྷ་མོ་ནི། །གངས་སེང་དཀར་མོ་གཡུ་རལ་ལ་ཆིབས་པ། །པུཎྜ་རི་ཀའི་མདོག་ཅན་ཞལ་གཅིག་པ། །ཕྱག་གཉིས་གསེར་གྱི་རྡོ་རྗེ་ཚེ་ཡི་བུམ། །དཔལ་བེའུ་གཡུང་དྲུང་དགའ་བ་འཁྱིལ་པས་མཚན། །ལྷ་རྫས་ཀཻ་ལ་ཤ་ཡི་ཁ་དོག་ཅན། །ན་བཟའ་ལྷབ་ལྷུབ་དབང་པོའི་གཞུ་འདྲ་བའི། །རིན་ཆེན་རྒྱན་རྣམས་ཀུན་གྱིས་མཛེས་པར་སྤྲས། །ཞི་འཛུམ་མཚར་སྡུག་ལང་ཚོའི་དཔལ་དུ་འབར། །མཁའ་འགྲོ་ཆེན་མོ་མཐིང་གི་ཞལ་བཟང་མ། །མི་གཡོ་བློ་བཟང་ཅོད་པན་མགྲིན་བཟང་དང༌། །གཏད་དཀར་འགྲོ་བཟང་ལ་སོགས་སྐུ་མདོག་ནི། །སྔོ་སེར་དམར་ལྗང་ཕྱག་ན་མེ་ལོང་དང༌། །བ་དན་ལྷ་ཡི་ཞལ་ཟས་ནེའུ་ལེ། །ནོར་བུ་རིན་ཆེན་སྒྲོམ་དང་དུར་ཝའི་རྩ། །ལྟོ་འགྲོའི་ཞགས་བསྣམས་རྒྱན་ལྡན་དར་བེར་གསོལ། །ཆིབས་སུ་རྐྱང་བུ་རྒྱ་སྟག་སྨན་ཤ་དང༌། །གཡུ་འབྲུག་སོགས་ལ་རོལ་པའི་སྟབས་ཀྱིས་བཞུགས། །ཚེ་རིང་མཆེད་ཀྱི་ལྷ་ལྔ་གསལ་བར་གྱུར། །གཞན་ཡང་མཁའ་འགྲོ་ཆེན་མོ་སོ་སོ་ཡི། །སྤྲུལ་པ་ཡང་སྤྲུལ་བྲན་དང་ཕོ་ཉའི་ཚོགས། །དཀར་ཕྱོགས་སྐྱོང་བའི་མཐུ་ལྡན་ཡུལ་ལྷ་རྣམས། །མ་ངེས་རྣམ་འགྱུར་སྣ་ཚོགས་ཅིར་ཡང་བསྒྱུར། །བསྟན་ལ་གུས་དང་རྣལ་འབྱོར་གྲོགས་མཛད་ཅིང༌། །ཕྲིན་ལས་འདོད་དགུར་བསྒྱུར་བའི་མཐུ་ལྡན་གྱུར། །ཀུན་གྱི་གནས་གསུམ་འབྲུ་གསུམ་ས་བོན་དང༌། །རང་གི་ཐུགས་ཀའི་ཧཱུྃ་ལས་འོད་ཟེར་འཕྲོས། །རྩ་གསུམ་དཀྱིལ་འཁོར་རབ་འབྱམས་རྒྱ་མཚོ་དང༌། །ཁྱད་པར་བསྐྱེད་པ་ལྟ་བུའི་ཌཱ་ཀིའི་ཚོགས། །དམ་ཚིག་གཉེན་པོའི་ཐ་ཚིག་དྲན་བཞིན་དུ། །རང་བཞིན་གནས་ནས་འཁོར་བཅས་སྤྱན་དྲངས་གྱུར། །

བདུག་སྤོས་འཐུལ་ཞིང་རོལ་མོ་ལྡིར་ལ། ཨོཾ་ཨཱཿཧཱུྃ། སྣང་སྲིད་ཆོས་སྐུའི་ཞིང་ཁམས་ནས། །བཀའ་བརྒྱུད་བླ་མས་བྱིན་ཕོབ་ཅིག །དག་པ་ལོངས་སྐུའི་ཞིང་ཁམས་ནས། །ཡི་དམ་ལྷ་ཚོགས་བྱིན་ཕོབ་ཅིག །གང་འདུལ་སྤྲུལ་པའི་ཞིང་ཁམས་ནས། །དཔའ་བོ་མཁའ་འགྲོའི་ལྷ་ཚོགས་དང༌། །བཀའ་དང་དམ་ཚིག་ཕོ་བྲང་ནས། །ཆོས་སྐྱོང་ཕོ་རྒྱུད་མོ་རྒྱུད་རྣམས། །གནས་འདིར་སྤྱན་འདྲེན་གཤེགས་སུ་གསོལ། །བྱིན་ཆེན་ཕོབ་ལ་དངོས་གྲུབ་སྩོལ། །སྣང་བ་དབང་སྡུད་ཕྱི་ནང་གི །བར་ཆད་རྐྱེན་བཟློག་ཕྲིན་ལས་བཞི། །མྱུར་དུ་འགྲུབ་ཕྱིར་གཤེགས་སུ་གསོལ། །འོག་མིན་བདེ་ཆེན་དག་པའི་ཞིང༌། །མ་ཚོགས་སྤྲིན་ལྟར་འདུ་བའི་གནས། །དུས་གསུམ་རྒྱལ་ཀུན་སྐྱེད་པའི་ཡུམ། །གཙོ་མོ་རྡོ་རྗེ་ཕག་ཞལ་ཅན། །སིངྷ་མུ་ཁའི་ཀློང་ཡངས་ནས། །སྒྱུ་འཕྲུལ་གར་གྱི་རོལ་རྩེད་ཀྱི། །དཔལ་འབར་སྤྲུལ་པའི་ལྷ་མོ་ཆེ། །དྷརྨ་ཏཱ་ར་འཁོར་བཅས་དང༌། །བཀྲ་ཤིས་ཚེ་རིང་ལྷ་མོ་ལྔ། །ཉི་མ་འབུམ་གྱི་གཟི་མདངས་ལྡན། །རྣལ་འབྱོར་དམ་ལྡན་སྒྲུབ་པ་པོས། །མཆོད་པའི་གནས་འདིར་སྤྱན་འདྲེན་ན། །དགྱེས་པའི་སྙན་འགྱུར་གླུ་སྒྲ་ལྡིར། །བརྩེ་བའི་སྤྱན་ཟུང་བུང་བ་གཡོ། །མ་ཐོགས་སྐད་ཅིག་ངང་ཉིད་ལ། །ནུབ་ཕྱོགས་ཨོ་རྒྱན་མཁའ་སྤྱོད་དང༌། །ཡུལ་ཆེན་སོ་གཉིས་དུར་ཁྲོད་བརྒྱད། །གང་དུ་བཞུགས་པའི་གནས་མཆོག་ནས། །མཁའ་འགྲོ་ཆེན་མོ་འཁོར་དང་བཅས། །བུ་ཡུག་འཚུབས་བཞིན་གཤེགས་སུ་གསོལ། །ཨོཾ་ཤྲཱི་མཱན་གུ་རུ་དེ་ཝ་དྷརྨ་ཏཱ་ར་ཌཱ་ཀི་ནཱི་ས་པ་རི་ཝཱ་ར་ཨི་ཛི་དྷ་རི་པཉྩ་ཌཱ་ཀི་བཛྲ་ས་མ་ཡ་ཛ་ཛ། ཞེས་སྤྱན་དྲངས་ཞིང་སྙན་པའི་དབྱངས་ཀྱིས་དྲངས་ཏེ་མདུན་བསྐྱེད་དམ་ཚིག་པར་སིབ་སིབ་མཚོ་ལ་ཆར་བབས་པ་བཞིན་དམིགས་ནས། ཛཿཧཱུྃ་བྃ་ཧོཿགཉིས་མེད་གྱུར། །ཧོ། འདིར་བཤམས་དར་དང་རིན་ཆེན་གྱི། །ཁྲི་མཆོག་དྲི་མེད་པདྨ་དཀར། །འདབ་སྟོང་གེ་སར་རྒྱས་པའི་སྟེང༌། །ཉི་ཟླའི་དཀྱིལ་འཁོར་གསལ་འཚེར་བར། །འཁོར་བཅས་བདེ་བར་བཞུགས་སུ་གསོལ། །ས་མ་ཡ་ཏིཥྛ་ལྷན།

ཨ་ཏི་པཱུ་ཧོ། དེ་ལྟར་གདན་འབུལ་དང་ཕྱག་འཚལ་ནས། ཧྲཱིཿ དབྱིངས་ཀྱི་ཡུམ་ཆེན་འཁོར་དང་བཅས། །སྔོན་ཚེ་དཀྱིལ་འཁོར་གཙོ་མཆོག་དཔལ། །རིག་འཛིན་པདྨ་སམ་བྷ་ཝའི། །བཀའ་དང་དམ་ཚིག་རྗེས་དགོངས་ལ། །རྩ་བ་བརྒྱུད་པའི་བླ་མ་དང༌། །མཉམ་མེད་འཇིག་རྟེན་གསུམ་མགོན་པོས། །རྡོ་རྗེའི་བཀའ་ལས་མ་འདའ་བར། །དེང་འདིར་བརྒྱུད་འཛིན་སྒྲུབ་པ་པོ། །བསྟན་དང་བསྟན་འཛིན་གང་ཟག་ལ། །སྲུང་སྐྱོབ་མི་འབྲལ་རྟག་པར་སྐྱོངས། །མ་ཧཱ་བཛྲ་ས་མ་ཡ། མ་ཏི་ཀྲ་མ་ཏི་ས་མ་ཡ་རཀྵནྟུ། ཞེས་སྔོན་གྱི་ཐ་ཚིག་གཡར་དམ་རྗེས་སུ་དྲན་ནས་རྣལ་འབྱོར་པའི་དམ་ལ་གནས་པར་མོས་ལ། མཆོད་པ་ནི། བྷྲཱུྃ་ལས་རཏྣའི་སྣོད་ཡངས་པར། །གཏོར་རྫས་སྨན་དང་རཀྟ་དང༌། །མཆོད་ཡོན་དག་བྱེད་ཀ་མ་ལ། །དྲི་ལྡན་མདངས་གསལ་བྱུགས་པའི་ངད། །རོ་བཅུད་སྙན་པའི་སིལ་ཁྲོལ་རྣམས། །རྩ་གསུམ་དཀྱིལ་འཁོར་ལྷ་ཡི་ཚོགས། །དྷརྨ་ཏཱ་ར་འཁོར་བཅས་དང། །བཀྲ་ཤིས་ཚེ་རིང་མཆེད་ལྔ་འཁོར། །སྤྲུལ་པར་བཅས་ལ་འབུལ་བར་བགྱི། །ཟག་མེད་བདེ་བ་ཆེན་པོར་རོལ། །གུ་རུ་དྷེ་ཝ་ཌཱ་ཀི་ནཱི་ས་པ་རི་ཝཱ་ར་ཨརྒྷཾ། པཱ་དྱཾ། པུཥྤེ། དྷུ་པེ། ཨཱ་ལོ་ཀེ། གནྡྷེ། ནེ་ཝི་དེ། ཤབྡ། པཉྩ་ཨ་མྲྀ་ཏ། མ་ཧཱ་རཀྟ། བ་ལིཾ་ཏ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། དེ་ནས་བསྟོད་པ་ནི། སྙན་པའི་དབྱངས་ཀྱི་གླུ་ཡིས་དྲངས་ཏེ། ཨོཾ་ཨཱཿཧཱུྃ་ཧོ། ཆོས་སྐུ་དྲི་མེད་མཁའ་ལྟར་ཁྱབ། །ལོངས་སྐུ་གསལ་མདངས་ལྷུན་གྱིས་གྲུབ། །སྤྲུལ་སྐུ་ཉིན་བྱེད་དབང་པོ་ལྟར། །གང་འདུལ་དེར་སྣང་པད་ཚལ་གྱི། །དགེ་ལེགས་འདབ་སྟོང་གྲོལ་བའི་གཉེན། །བཀའ་བརྒྱུད་བླ་མ་ཡི་དམ་དང༌། །ཆོས་སྐྱོང་སྲུང་མའི་ཚོགས་བཅས་བསྟོད། །ཛྙཱ་ན་ཌཱ་ཀི་ཡོངས་ཀྱི་གཙོ། །དབྱིངས་ཕྱུག་མ་མོ་འཁོར་བཅས་དང༌། །ཞིང་སྐྱོང་མཁའ་འགྲོ་མཆེད་ཟླ་ལྔ། །སྤྲུལ་དང་ཡང་སྤྲུལ་དཀར་ཕྱོགས་སྐྱོང༌། །མཐུ་ལྡན་རྣམས་ལ་བསྟོད་པར་བགྱི། །གྲུབ་པ་བརྒྱད་དང་ཕྲིན་ལས་བཞི། །མཆོག་དང་ཐུན་མོང་ལ་སོགས་པ། །གང་བཅོལ་ཐོགས་མེད་ཉིད་དུ་སྒྲུབས། །

དེ་ནས་བཟླས་པ་བྱ་བ་ནི། མདུན་བསྐྱེད་གཙོ་མོའི་ཐུགས་ཀ་རུ། །ཟླ་སྟེང་ཧྲཱིཿཡིག་དཀར་པོ་གསལ། །མཁའ་འགྲོ་བཞི་ཡི་ཐུགས་ཀར་ཡང༌། །ཧ་རི་ནི་སས་མཚན་པར་གྱུར། །ཧྲཱིཿམཐར་སྔགས་ཕྲེང་མུ་ཏིག་མདོག །གཡོན་དུ་འཁོར་ལས་འོད་འཕྲོས་པས། །མཆོག་གསུམ་རྩ་གསུམ་མཆོད་པས་མཉེས། །སེམས་ཅན་ཀུན་གྱི་སྒྲིབ་གཉིས་སྦྱངས། །ཐུགས་དམ་རྒྱུད་བསྐུལ་འོད་ཟེར་གྱིས། །ཕྱོགས་བཅུའི་རྒྱལ་བའི་བྱིན་རླབས་དང༌། །རྩ་གསུམ་རིག་འཛིན་དཔའ་བོ་དང༌། །ཌཱ་ཀི་དམ་ཅན་མཐུ་ནུས་སྟོབས། །སྐུ་དང་འབྲུ་གསུམ་ཕྱག་མཚན་དང༌། །འོད་ཀྱི་རྣམ་པར་སྤྱན་དྲངས་ནས། །རང་དང་མདུན་བསྐྱེད་ལ་ཐིམ་གྱུར། ། སྲོག་སྔགས་ནི། ཨོཾ་ཛྙཱ་ཎ་ཌཱ་ཀི་ནཱི་ཡེ་ཧ་རི་ནི་ས་ཧཱུྃ། རྩ་བའི་སྔགས་ནི། ཨོཾ་གྷ་མུ་ཨ་ཙི་ཏ་ཤྭ་ར་དྷརྨ་ཏཱ་ར་མ་མ་ཙཀྲ་སྭཱ་ཧཱ། ཡར་འདུས། སརྦ་འདུས། མ་མ། ཛྙཱ་ན་ཏཱ་ར་འདུས། གྷ་མུ་ས་མ་ཡ་ཏཱ་ར་དེ་མ་ཌཱ་ཀི་ནི་སརྦ་མ་མ་འདུས་འདུས་ཧཱུྃ་ཕཊ། དེ་ལྟར་སྔགས་ཕྲེང་གསལ་བར་བསྒོམ། །སྤྲོ་བསྡུ་དང་པོས་རྒྱལ་རྣམས་མཆོད། །གཉིས་པས་བྱིན་རླབས་དངོས་གྲུབ་བསྡུ། །སྙིང་པོ་ཅི་ནུས་བགྲངས་པའི་མཐར། །སྤྲོ་བསྡུ་གསུམ་པ་འདི་ལྟར་དམིགས། །མདའ་དར་གཡབ་ཅིང་སྤོས་བསྲེགས་ལ། ། སླར་ཡང་བདག་མདུན་ཐུགས་ཀ་ནས། །འོད་འཕྲོས་ལས་བཞི་མཁའ་འགྲོ་དང༌། །བཀྲ་ཤིས་ཚེ་རིང་འཁོར་བཅས་ཀྱི། །ཐུགས་རྒྱུད་བསྐུལ་ཏེ་འོད་ཟེར་དང༌། །རང་རང་ཕྱག་མཚན་སྤྲོས་པ་ལས། །འབྱུང་ལྔའི་བཅུད་དང་སྲིད་གསུམ་གྱི། །ལྷ་ཀླུ་གནོད་སྦྱིན་མི་མ་ཡིན། །མི་བདག་འབྱོར་པས་དྲེགས་རྣམས་ཀྱི། །ཚེ་དང་བསོད་ནམས་དཔལ་དང་གྲགས། །དབང་ཐང་ཟས་ནོར་ལོངས་སྤྱོད་ཀུན། །ཁབ་ལེན་རྡོ་ཡིས་ལྕགས་བསྡུས་ལྟར། །འོད་ཟེར་ཕྱག་མཚན་རྣམས་ལ་བསྡུས། །ཁམས་གསུམ་གླིང་བཞི་ཁོར་ཡུག་བཅས། །དབང་སྡུད་མངའ་དབང་བསྒྱུར་བར་གྱུར། །ཀུན་གྱི་སྙིང་པོའི་ཡང་བཅུད་རྣམས། །བདག་དང་མདུན་བསྐྱེད་གཙོ་མོ་ལ། །འབུལ་བཞེས་དཔལ་འབྱོར་འདོད་དགུའི་ཆར། །དངོས་གྲུབ་མཚོ་ལ་ཆར་བབས་གྱུར། །ཨོཾ་གྷ་མུ་ཨ་ཙི་ཏ།

ཤྭ་ར་དྷརྨ་ཏཱ་ར། མ་མ་ཙཀྲ་སྭཱ་ཧཱ། ཡར་འདུས། སརྦ་འདུས། རཱ་ཛ་རཱ་ཛ་འདུས། མ་མ་ཛྙཱ་ན་ཏཱ་ར་འདུས། གྷ་མུ་ས་མ་ཡ་ཏཱ་ར། དེ་མ་ཌཱ་ཀི་ནི་ཡེ། མ་མ་ཨཱ་ཡུ་པུཎྱེ་པུཥྚིཾ་ལོ་ཀ་ཝ་ཤང་ཀུ་རུ་སརྦ་རཏྣ་སིདྡྷི་ཕ་ལ་ཛཿཧཱུྃ། ཞེས་འདོད་གསོལ་སྤེལ་ཚིག་ཅན་གྱི་བསྙེན་པ་ལ་འབད་པར་བྱའོ། །འཁོར་རྣམས་འོད་ཟེར་སྤྲོ་བསྡུ་ལ་འདྲེན་པ་ཙམ་ལས་སྔགས་ཟུར་དུ་མེད་པ་ནི། གཙོ་མོ་ཉིད་ལ་གནད་དུ་བསྣུན་པས་འཁོར་རྣམས་ཤུགས་ཀྱིས་འགྲུབ་པར་འགྱུར་བ་ནི་གནད་ཐུན་མོང་མ་ཡིན་པའོ། །འདི་ཉིད་འགྲུབ་པའི་རྟགས་ནི། བུད་མེད་ཆས་བཟང་གིས་ཟས་ནོར་སྟེར་བ་དང༌། ཉི་མ་འཆར་བ་དང༌། རི་ལ་ཕྱིན་པ། མེ་ཏོག་འཐུ་བ། འཇའ་ཤར། འབྲུ་ཡི་རིགས་སྨིན་པ་དང་སྐྱེ་བ་རྨིས་ན་དངོས་གྲུབ་ཉེ་བའི་ལྟས་དང༌། སེམས་འཚུབས། ཏིང་འཛིན་མི་གསལ། བྱིངས་རྨུགས། རྨི་ལམ་དུ་བུད་མེད་ཁྲོ་བ། གོས་ངན་ཅན་དང་མཛེ་མོའི་གཟུགས་དང་མཚན་མ་སྟོན་པ་སོགས་བྱུང་ན། ཉམས་གྲིབ་ཀྱི་རྟགས་དང་མཆོད་པ་དམན་པ་ཡིན་པས། མདུན་བསྐྱེད་ཀྱི་སྤྱི་བོར་རྡོར་སེམས་བསྐྱེད་ལ་ཡིག་བརྒྱ་བཟླ། འདི་ཉིད་ཀྱི་སྐོང་བ་དང་བཤགས་པ་ནན་ཏན་དུ་འབད་པར་རིགས་སོ། །དེ་ནས་མཆོད་པའི་རིམ་པ་ལ། །བྱིན་རླབས་བསངས་སྦྱངས་སྤེལ་བ་རྣམས། །སྔོན་དུ་སོང་ཞིང་མཆོད་པའི་རྫས། །ཡིད་འོང་ཀུན་ཏུ་བཟང་པོ་ཡི། །སྨོན་ལམ་ཇི་བཞིན་བསྐྱེད་ལ་ཕུལ། ། ཨོཾ་ཨཱཿཧཱུྃ། ཆོས་སྐུ་ཀུན་ཏུ་བཟང་པོ་དང༌། །ལོངས་སྐུ་ཁྱབ་བདག་རྡོ་རྗེ་འཆང༌། །སྤྲུལ་སྐུ་རྒྱལ་དབང་པདྨ་འབྱུང༌། །ཕྱོགས་བཅུའི་ཞིང་ཁམས་མཐའ་ཡས་ནས། །སངས་རྒྱས་བྱང་སེམས་སྲས་དང་བཅས། །རྣལ་འབྱོར་དབང་ཕྱུག་ཏེ་ལོ་དང༌། །ནཱ་རོ་མར་མི་སྒམ་པོ་པ། །དུས་མཁྱེན་ཕག་གྲུ་རས་ཆེན་ཞབས། །ཀརྨ་པ་སོགས་ཀཾ་ཚང་གི །སྙིང་པོ་དོན་གྱི་བརྒྱུད་རྣམས་དང༌། །ཨ་མེས་ཚུལ་ཁྲིམས་རྒྱ་མཚོ་དང༌། །སྲས་བཞི་ཀྱུ་རའི་གདུང་བརྒྱུད་ཀྱི། །སྐྱེས་མཆོག་འཇིག་རྟེན་གསུམ་མགོན་པོ། །འབྲི་གུང་རིན་ཆེན་སྒྲུབ་བརྒྱུད་ཀྱི། །གྲུབ་བརྙེས་བླ་མ་རིམ་བྱོན་དང༌། །བདེ་དགྱེས་གསང་འདུས་མཱ་ཡཱ་དང༌། །རྣལ་འབྱོར་མ་སོགས་རྒྱུད་སྡེ་བཞིའི། །རབ་འབྱམས་དཀྱིལ་འཁོར་ལྷ་ཡི་ཚོགས། །གནས་གསུམ་སྐྱོང་བའི་དཔའ་བོ་དང༌། །ཌཱ་ཀི་ཆོས་སྲུང་དཔལ་མགོན་པོ། །ཡབ་གཅིག་ལེགས་ལྡན་བིང་བེར་ནག

ཨོཾ་ཤྲཱི་མཱན་ སོགས། བཛྲ་དྷུ་པེ་པྲ་ཏཱིཙྪ་ཡེ་སྭཱ་ཧཱ། དཔལ་མགོན༴ ཆོས་སྐྱོང་༴ མ་ཆོས་༴ སྣང་གསལ་བྱེད་ཉི་ཟླའི་འོད་ལྟར་འཚེར། །དགྱེས་པ༴ ཨོཾ་ཤྲཱི་མཱན་ སོགས། བཛྲ་ཨཱ་ལོ་ཀེ་པྲ་ཏཱིཙྪ་ཡེ་སྭཱ་ཧཱ། དཔལ་མགོན༴ ཆོས་སྐྱོང་༴ མ་ཆོས་ཀྱི༴ དྲི་གུར་གུམ་གེ་སར་བྱུགས་པའི་ཆབ། །དགྱེས་པ་༴ ཨོཾ་ཤྲཱི་མཱན་ སོགས། བཛྲ་གནྡྷེ་པྲ་ཏཱིཙྪ་ཡེ་སྭཱ་ཧཱ། དཔལ་མགོན༴ ཆོས་སྐྱོང་༴ མ་ཆོས་ཀྱི༴ བཅུད་ལྷ་ཡི་ཞལ་ཟས་རོ་བ རྒྱ་ལྡན། །དགྱེས་པ༴ ཨོཾ་ཤྲཱི་མཱན་ སོགས། བཛྲ་ནཻ་ཝིདྱ་པྲ་ཏཱིཙྪ་ཡེ་སྭཱ་ཧཱ། དཔལ་མགོན་གྱུར༴ ཆོས་སྐྱོང་༴ མ་ཆོས༴ སྒྲ་སྙན་པའི་རིག་བྱེད་སྣ་ཚོགས་ལྡིར། །དགྱེས་པ༴ ཨོཾ་ཤྲཱི་མཱན་ སོགས། བཛྲ་ཤབྡ་པྲ་ཏཱིཙྪ་ཡེ་སྭཱ་ཧཱ། རོལ་མོའི་རིགས་འབུད་བརྡུང་དཀྲོལ་བར་བྱའོ། །འདོད་ཡོན་ལྔ་འབུལ་བ་ནི། རྨད་བྱུང་མཚར་སྡུག་ལང་ཚོ་བཀྲ། །སྙན་དབྱངས་ཀ་ལ་པིངྐའི་འགྱུར། །གདུང་སེལ་ཧ་རི་ཙནྡན་དྲི། །རྩི་བཅུད་དྭངས་མའི་ཟས་མཆོག་བཤམས། །དབང་གཞུ་ལྟར་མཛེས་ལྷའི་རེག་བྱ། །འདོད་ཡོན་ལྔ་ཡི་མཆོད་པ་ཡང༌། །འབུལ་གྱི་དགྱེས་བརྩེས་རོལ་པར་མཛོད། །ཨོཾ་ཤྲཱི་མཱན་ སོགས་སྔགས་གོང་དང་འདྲ། བཛྲ་རཱུ་པ། ཤབྡ། གནྡྷེ། ར་ས། སྤརྴེ་པྲ་ཏཱིཙྪ་ཡེ་སྭཱ་ཧཱ། དེ་ནས་བཀྲ་ཤིས་པའི་རྟགས་བརྒྱད་འབུལ་བ་ནི། རིན་ཆེན་མུ་ཏིག་ཟ་འོག་བརྒྱན་པ་ཡི། །ནོར་བུའི་ཏོག་ཅན་གདུགས་བཟང་གསེར་ཉ་འཁྱུགས། །དགོས་འདོད་འབྱུང་བའི་བུམ་པ་ཆུ་སྐྱེས་ཀྱི། །པདྨ་འདབ་སྟོང་སྐྱེ་བ་ལྔ་པའི་དུང༌། །དཔལ་བེའུ་མི་ནུབ་རྒྱལ་མཚན་ལྷབ་ལྷུབ་གཡོ། །གསེར་གྱི་འཁོར་ལོ་བཀྲ་ཤིས་རྟགས་བརྒྱད་འབུལ། །ཨོཾ་ཤྲཱི་མཱན་ སོགས། ཨཥྚ་མངྒ་ལམ་ཧེ་ཏུ་པྲ་ཏཱིཙྪ་ཡེ་སྭཱ་ཧཱ། བཀྲ་ཤིས་པའི་རྫས་བརྒྱད་འབུལ་བ་ནི། སྒྲིབ་མེད་རྙོག་པ་དང་བྲལ་མེ་ལོང་རབ་ཏུ་གསལ། །ཉོན་མོངས་དུག་གསུམ་འཆིང་བ་སེལ་བྱེད་གི་ཧཾ་དང༌། །རྩི་དང་བཅུད་ཀྱི་སྙིང་པོ་རྣམ་པར་དག་གཙང་ཞོ། །ཚེ་གྲངས་མེད་པར་འཕེལ་འགྱུར་སྔོ་འཇམ་དུར་ཝའི་རྩྭ། །ཡིད་ལ་འདོད་འཇོ་ལྷུན་གྱིས་གྲུབ་སྟེར་བིལ་བའི་འབྲས། །ཆོས་ཀྱི་སྒྲ་དབྱངས་སྲིད་ན་ལྡིར་བའི་གཡས་འཁྱིལ་དུང༌། །བརྟན་གཡོ་མ་ལུས་དབང་དུ་སྤེལ་བྱེད་ལི་ཁྲི་སྟེ། །བདུད་བཞིའི་དཔུང་གཞོམ་ཡུངས་ཀར་བཀྲ་ཤིས་རྫས་མཆོག་རྣམས། །འབུལ་གྱི་ཐུགས་རྗེས་བརྩེ་དགོངས་དགྱེས་པར་རོལ་པར་མཛོད། །ཨོཾ་ཤྲཱི་མཱན་

སོགས། ཨཥྚ་མངྒ་ལམ་དྲ་བྱ་པྲ་ཏཱིཙྪ་ཡེ་སྭཱ་ཧཱ། རྒྱལ་སྲིད་སྣ་བདུན་འབུལ་བ་ནི། གླིང་བཞིར་དབང་བསྒྱུར་འཁོར་ལོ། །དགོས་འབྱུང་ཡིད་བཞིན་ནོར་བུ། །བདེ་སྟེར་བཙུན་མོ་རིན་ཆེན། །ཟབ་ཡངས་འཕྲུལ་ལྡན་མི་བདག །སྟོབས་བཅུ་ལྡན་པ་གླང་པོ། །རྫུ་འཕྲུལ་རྩལ་ལྡན་རྟ་མཆོག །ཕས་རྒོལ་ཚར་གཅོད་དམག་དཔོན། །རྒྱལ་སྲིད་རིན་ཆེན་བདུན་ཡང༌། །འབུལ་གྱི་དགྱེས་བརྩེར་རོལ་ཅིག །ཨོཾ་ཤྲཱི་མཱན་ སོགས། སབྟ་རཏྣ་པྲ་ཏཱིཙྪ་ཡེ་སྭཱ་ཧཱ། ཞེས་ཕྱི་མཆོད་ཀྱི་རིམ་པ་རྣམས་གྲུབ་ནས། ནང་མཆོད་སྨན་གྱི་བདུད་རྩི་འབུལ་བ་ནི། ཨོཾ། ཐོད་པ་མཚན་ལྡན་ཡངས་པའི་ནང༌། །རྩ་བརྒྱད་ཡན་ལག་སྟོང་སྦྱར་བའི། །སྨན་མཆོག་ཡེ་ཤེས་བདུད་རྩི་དངོས། །འབུལ་གྱི་དགྱེས་བརྩེར་དགོངས་ཤིང་རོལ། །ཨོཾ་ཤྲཱི་མཱན་ སོགས། པཉྩ་ཨ་མྲི་ཏ་ཁ་རཾ་ཁཱ་ཧི། རཀྟའི་མཆོད་པ་འབུལ་བ་ནི། ཨཱཿ ལྕང་ལོར་བཅས་པའི་ཀ་པཱ་ལར། །རྒྱུ་དྲུག་བསྒྲལ་བའི་དམར་ཆེར་བསྐྱིལ། །ཆགས་མེད་རཀྟའི་རྦ་རླབས་སྦྲེངས། །འབུལ་གྱི་དགྱེས་བརྩེར་དགོངས་ཤིང་རོལ། །ཨོཾ་ཤྲཱི་མཱན་ སོགས་དང༌། མ་ཧཱ་རཀྟ་ཁ་རཾ་ཁཱ་ཧི། གཏོར་མའི་མཆོད་པ་ནི། ཧཱུྃ། ཨཱིནྡྲ་ནཱི་ལའི་པད་སྣོད་ནང༌། །འབྲུ་བཅུད་རིན་ཆེན་སྨན་སྦྱར་བའི། །དཀར་དམར་རྒྱན་མཛེས་བདུད་རྩིས་བྲན། །གཏོར་ཆེན་འབུལ་གྱི་དགྱེས་བརྩེར་རོལ། །ཨོཾ་ཤྲཱི་མཱན་ སོགས་དང༌། ། མ་ཧཱ་བ་ལིངྟ་ཁ་རཾ་ཁཱ་ཧི། དེ་ནས་གསང་མཆོད་འབུལ་བ་ནི། ཧྲཱིཿ ཐབས་ཤེས་གཉིས་མེད་ཟུང་འཇུག་སྦྱོར། །བདེ་ཆེན་དགའ་བཞི་ཡེ་ཤེས་དངོས། །བླ་མེད་གསང་བའི་མཆོད་པ་སྟེ། །ཟག་མེད་བདེ་བའི་ངང་དུ་རོལ། །ཨོཾ་ཤྲཱི་མཱན་ སོགས་དང༌། མ་ཧཱ་སུ་ཁ་ཨཱཿཧཱུྃ་སྭཱ་ཧཱ། དེ་ནས་དེ་ཁོ་ན་ཉིད་ཀྱི་མཆོད་པ་ཆེན་པོ་ནི། ཧོ། ཆོས་ཀུན་གདོད་ནས་སྟོང་པ་ལ། །བློས་བྱས་གཉིས་འཛིན་ཆོས་རྣམས་ཀྱང༌། །ཡེ་ཤེས་རང་སྣང་རྩལ་ལས་གྲུབ། །མ་འགགས་རང་གསལ་ངང་དུ་འབུལ། །ཨོཾ་ཤྲཱི་མཱན་ སོགས་ནས།

དྷརྨ་དྷརྟུ་ཤུནྱཱ་ཏ་ཛྙཱ་ན་གརྦྷ་ཨཿཧཱུྃ། དེ་ལྟར་མཆོད་པ་ཕུལ་བ་ཡིས། །རྩ་གསུམ་སྐུ་གསུང་ཐུགས་མཉེས་ནས། །བདག་གཞན་བསོད་ནམས་ཚོགས་རྫོགས་ཤིང༌། །མ་ཚོགས་མཆོད་པས་དགྱེས་གྱུར་ནས། །རྣམ་བཞིའི་ཕྲིན་ལས་དངོས་གྲུབ་ཀྱི། །ཆར་གྱི་དཔལ་འབྱོར་རྒྱས་པར་བསམ། །དེ་ནས་ལྷ་ཚོགས་མ་ལུས་པར། །སྙན་དབྱངས་ཡན་ལག་རྒྱ་མཚོ་ཡིས། །ཕྱག་འཚལ་བསྟོད་པར་འཇུག་པ་ནི། ཨོཾ་སྐྱེ་འགགས་གནས་བྲལ་རང་བཞིན་ཀློང་ཡངས་ནས། །སངས་རྒྱས་བྱང་སེམས་བླ་མ་ཡི་དམ་དང༌། །དཔའ་བོ་མཁའ་འགྲོ་ཆོས་ཀྱི་སྲུང་མའི་ཚོགས། །ཐུགས་རྗེའི་སྣང་བྱེད་འོད་ཀྱིས་ཞིང་ཀུན་ཁྱབ། །རྨད་བྱུང་བྱིན་རླབས་སྤྲིན་ཕུང་ཐིབས་སེ་ཐིབ། །མཆོག་ཐུན་དངོས་གྲུབ་ཆར་རྒྱུན་སི་ལི་ལི། །གུས་པས་ཐལ་སྦྱར་ཀུ་མུཏ་ཟུམ་བྱས་ནས། །རྩ་གསུམ་དཀྱིལ་འཁོར་ལྷ་ལ་ཕྱག་འཚལ་ལོ། །དོན་དམ་ཆོས་ཀྱི་དབྱིངས་ཀྱི་ངང་ཉིད་ལས། །མ་འགགས་རང་རྩལ་སྒྱུ་འཕྲུལ་གར་སྟོན་པའི། །དུ་མ་རོ་གཅིག་སྣང་སྟོང་བདེ་ཆེན་སྐུ། །རྡོ་རྗེ་རྣལ་འབྱོར་མ་ལ་ཕྱག་འཚལ་བསྟོད། །གདོད་ནས་དོན་གྱི་ཡེ་ཤེས་མངོན་གྱུར་ཀྱང། །སྙིང་རྗེས་འགྲོ་ལ་རྗེས་ཆགས་བྱང་སེམས་བློས། །ཐབས་མཁས་གང་འདུལ་སྤྲུལ་པའི་སྐུར་བཞེངས་མ། །བཅོམ་ལྡན་དྷརྨ་ཏཱ་རར་ཕྱག་འཚལ་བསྟོད། །ཐུགས་རྗེ་གཏིང་མཐའ་མེད་པའི་རྒྱ་མཚོ་ལས། །སྲིད་ཞིའི་གདུང་སེལ་རིག་འཛིན་ཡིད་བཞིན་དབང༌། །དགོས་འདོད་བྱང་ཆུབ་ནོར་གྱི་མཆོག་རྣམས་ཀྱི། །སྐྱེད་པའི་ཡུམ་གྱུར་མ་ལ་ཕྱག་འཚལ་བསྟོད། །ལྟ་བས་མི་ངོམས་གཟུགས་སྐུ་མཚན་བཟང་བཀྲ། །ཚངས་པའི་དབྱངས་ལྡན་ཟབ་ཞི་སྤྲོས་བྲལ་ཐུགས། །ཚད་མེད་རྣམ་བཞིས་འགྲོ་བའི་དོན་མཛད་མ། །ཨ་ཕྱི་ཆོས་ཀྱི་སྒྲོལ་མར་ཕྱག༴ ཉོན་མོངས་ལས་ཀྱི་ནད་གདོན་ཞི་མཛད་ཅིང༌། །ཚེ་དང་བསོད་ནམས་དཔལ་འབྱོར་རྒྱས་བྱེད་མ། །སྣང་སྲིད་དབང་སྡུད་གདུག་པའི་དགྲ་བགེགས་བསྒྲལ། །མཆེད་ཟླ་མཁའ་འགྲོ་སྡེ་བཞིར་ཕྱག༴

དཔུང་གི་ཤུགས་འཆང་ཟ་བྱེད་དཔའ་མོའི་གཙོ། །བསྟན་དང་བསྟན་འཛིན་ཕྲིན་ལས་སྤེལ་བའི་གྲོགས། །རིགས་ཀྱི་ཡུམ་ཆེན་ལྔ་ལ་ཕྱག༴ དམ་ཚིག་རྗེས་གཅོད་བཀའ་ཡི་བྱ་ར་བ། །དབང་མོ་ཉེར་བརྒྱད་བརྟན་མ་སྐྱོང་རྣམས་དང༌། །ཟ་གསོད་འཕྲོག་བྱེད་ཕོ་ཉ་ཤན་པའི་ཚོགས། །མཐུ་དང་རྩལ་ལྡན་རྣམས་ལ་བསྟོད་པར་བགྱི། ། དེ་ལྟར་བསྟོད་ནས། རྩ་གསུམ་དཀྱིལ་འཁོར་ལྷ་ཚོགས་ལ། །དགྱེས་མཆོད་ཁྱད་པར་རྨད་བྱུང་བ། །གསང་པ་དམ་རྫས་ཚོགས་འཁོར་ལོ། །འབུལ་བའི་རིམ་པ་བསྟན་པར་བྱ། །དང་པོ་ཚོགས་སྦྱངས་སྤེལ་ནི། ཧྲཱིཿ རང་ཉིད་བླ་མར་གསལ་བ་ཡི། །ཐུགས་ལས་འོད་ཟེར་སྐར་ལྟར་འཕྲོས། །ཞལ་ཟས་འདོད་ཡོན་ཚོགས་ལ་ཐིམ། རྃ་ཡྃ་ཁྃ་གིས་བསྲེག་གཏོར་བཀྲུས། །སླར་ཡང་ཨོཾ་ཨཱཿཧཱུྃ་འཕྲོས་ནས། །ཟག་མེད་ཡེ་ཤེས་བདུད་རྩིར་བསྒྱུར། །ཨོཾ་ཨཱཿཧཱུྃ་ཧ་ཧོ་ཧྲཱིཿ ལན་གསུམ་གྱིས་བྱིན་བརླབས་ལ། ཚོགས་དང་པོ་འབུལ་བ་ནི། ཧྲཱིཿ བླ་མེད་བླ་མ་རྒྱལ་དང་སྲས། །ཡི་དམ་ཆོས་སྐྱོང་ཌཱ་ཀིའི་ཚོགས། །དམ་རྫས་ཚོགས་ཀྱི་འཁོར་ལོ་ཡི། །མཆོད་བཞེས་དབང་དང་དངོས་གྲུབ་སྩོལ། །གྷ་ན་ཙཀྲ་སརྦ་པཱུ་ཛ་ཁ་ཁ་ཁཱ་ཧཱི། ཞེས་ཚོགས་ཕུད་ཕུལ་ནས། བར་པ་བསྐང་བཤགས་ནི། ཚོགས་དང་བསྐང་གཏོར་མཆོད་པ་རྣམས་གསོ་བྲན་ལེགས་པར་བྱས་ཏེ། སྨན་གཏོར་རཀྟ་སྦྲེང་ཞིང་སྤོས་བསྲེག །རོལ་མོ་ལྡིར་ལ་འདི་སྐད་དོ། ། ཧྲཱིཿ ཕྱི་སྣོད་སྣང་སྲིད་རབ་འབྱམས་ཞིང༌། །ཀུན་བཟང་མཆོད་པའི་སྤྲིན་སྤྲོ་བའི། །སྔགས་དང་ཕྱག་རྒྱས་བྱིན་བརླབས་ནས། །ལྷ་ཚོགས་དགྱེས་པ་སྐོང་བྱེད་གྱུར། །ཨོཾ་སརྦ་དྷརྨ་སུ་ཙ་པཉྩ་རཏྣ་ཀ་ཡ་གོ་ཏེ་བི་བྷ་བ་ཀཿསྭ་བྷཱ་ཝ་ཤུདྡྷ྅ཾ། ཧྲཱིཿ ཡེ་ནས་མ་བཅོས་རང་བྱུང་སྤྲོས་བྲལ་དབྱིངས། །ཆོས་སྐུ་ཀུན་ཏུ་བཟང་པོའི་ཐུགས་དམ་བསྐང། །རང་མདངས་མ་འགགས་རྩལ་ལས་ལོངས་སྤྱོད་སྐུ། །ལྔ་ལྡན་རྡོ་རྗེ་འཆང་ཆེན་ཐུགས༴ གང་འདུལ་སྤྲུལ་བསྒྱུར་གར་གྱི་རོལ་རྩེད་ཅན། །ཨོ་རྒྱན་པདྨཱ་ཀ་རའི༴

ཕྱོགས་བཅུའི་ཞིང་བཞུགས་དུས་གསུམ་འགྲོ་ལ་གཟིགས། །སངས་རྒྱས་བྱང་སེམས་རྣམས་ཀྱི༴ རབ་འབྱམས་ཡེ་ཤེས་དཀྱིལ་འཁོར་ཀློང་ཡངས་ནས། །ཡི་དམ་རྒྱུད་སྡེ་བཞི་ཡི༴ གནས་གསུམ་རང་བཞིན་དག་པའི་ཞིང་ཁམས་ནས། །གྲུབ་ཆེན་དཔའ་བོ་མཁའ་འགྲོའི་ཐུགས་དམ་བསྐང༌། །ཡེ་ཤེས་ལས་དང་འཇིག་རྟེན་ཆོས་སྲུང་མ། །དྲེགས་བྱེད་དམ་ཅན་རྒྱ་མཚོའི་ཐུགས་དམ་བསྐང༌། །བྷྱོ། བཅོམ་ལྡན་རྡོ་རྗེ་བཙུན་མོའི་རྣམ་པར་འཕྲུལ། །རིག་འཛིན་གསང་བའི་དགྱེས་སྡེ་དབྱིངས་ཕྱུག་མ། །བསྟན་སྐྱོང་མཁའ་འགྲོའི་གཙོ་མོ་དཔལ་འབར་མ། །མ་གཅིག་དྷརྨ་ཏཱ་རའི་ཐུགས་དམ་བསྐང༌། །ཡེ་ཤེས་ཌཱ་ཀི་གཙུག་ན་ནོར་བུ་དང༌། །དམ་ཚིག་མཁའ་འགྲོ་ཆེན་མོ་ཡིད་བཞིན་ནོར། ཞེས་འབུལ་ནས་བཤལ་ཆུ་ཤིན་ཏུ་ལྷག་མ་ལ་དབང་བ་རྣམས་ལ་སྦྱིན་ནོ། །དེ་ནས་ཐུགས་དམ་བསྐུལ་བ་ནི། བྷྱོ། མཁའ་དབྱིངས་ཨཱིནྡྲ་ནཱི་ལའི་མདངས་འཚེར་བའི། །མཐིང་གསལ་འོད་འབར་སྲ་བརྟན་གཡུ་ཡི་བྲག །གཏེར་སྒྲོམ་ཤེལ་གྱི་ཕོ་བྲང་ནང་ཤེད་ནས། །ཆེ་ཞེས་སྲིད་པའི་རྒྱལ་མོ་མཐུ་ཆེན་མ། །རིག་འཛིན་རྒྱུད་ཀྱི་སྲུང་མ་ཐུགས་དམ་ཅན། །རྣལ་འབྱོར་དགྲ་ལྷ་ཆེན་མོ་འཁོར་དང་བཅས། །རྒྱལ་བའི་བསྟན་པ་བསྲུང་ཕྱིར་དབྱིངས་ནས་བསྐུལ། །སྒྲུབ་པའི་བར་ཆད་བསལ་ཕྱིར་དབྱིངས༴ ལས་བཞིའི་ཕྲིན་ལས་བཅོལ་ཕྱིར༴ བྱད་མ་རྦོད་གཏོང་བཟློག་ཕྱིར༴ འདིར་གཤེགས་དམ་རྫས་གཏོར་མ་འདི་བཞེས་ལ། །འཁོར་བཅས་ཐུགས་དམ་གཉན་པོའི་དུས་ལ་བབས། །དཀར་ཕྱོགས་སྡེ་དཔུང་ཀུན་གྱི་གཙོ་མོ་སྟེ། །བཀྲ་ཤིས་ཚེ་རིང་མཆེད་ལྔ་འཁོར་བཅས་ཀྱང༌། །བསྙེན་ཅིང་སྒྲུབ་པའི་སྲུང་མ་གཡར་དམ་ཅན། །ཡེ་ཤེས་ལས་དང་འཇིག་རྟེན་ཌཱ་ཀིའི་ཚོགས། །སངས་རྒྱས་བསྟན་སྲུང་རྣལ་འབྱོར་ཚེ་དཔལ་སྤེལ། །རྐྱེན་ངན་བར་ཆད་བཟློག་ཕྱིར་དབྱིངས༴ བསྟན་དགྲ་སྒོས་དགྲ་བསྒྲལ་ཕྱིར་དབྱིངས་ནས༴ འདིར་གཤེགས་བརྒྱན་པའི་གཏོར་མ་འདི་བཞེས་ལ། །བཅོལ་བའི་ཕྲིན་ལས་གཡེལ་མེད་བསྒྲུབ་པར་མཛོད། ། ཅེས་བསྐུལ་ཏེ་དམར་གཏོར་གཉིས་དབུལ་བར་བྱའོ། །དེ་ནས་ཆད་ཐོ་ནི། བྷྱོཿ སྔོན་ཚེ་ཆོས་དབྱིངས་དག་པའི་དཀྱིལ་འཁོར་དུ། །བཅོམ་ལྡན་ཀུན་ཏུ་བཟང་མོའི་སྤྱན་སྔ་རུ། །བླ་མེད་དོན་དམ་བྱང་ཆུབ་སེམས་བསྐྱེད་ལྟར། །ཚད་མེད་སྙིང་རྗེ་ཆེན་པོས་བརྩེར་དགོངས་ནས། །མཆོག་ཐུན་དངོས་གྲུབ་སྩོལ་བར་མ་ཆད་དམ། །སྣོད་བཅུད་བྱིན་རླབས་འབེབས་པར་མ་ཆད་དམ། །དམ་རྫས་བདུད་རྩིའི་གཏོར་མ་འདི་བཞེས་ལ། །རྣལ་འབྱོར་བཅོལ་བའི་ཕྲིན་ལས་བསྒྲུབ་པར་མཛོད། །དེ་འོག་གནས་མཆོག་གཏེར་སྒྲོམ་ཤེལ་བྲག་ཏུ། །ཨོ་རྒྱན་ཆེན་པོ་པདྨ་ཐོད་ཕྲེང་གི །སྤྱན་སྔར་ལས་ཀྱི་མ་མོའི་སྐུར་བསྟན་ནས། །གསང་སྔགས་དཀྱིལ་འཁོར་སྐྱོང་བར་གཡར་དམ་བཅས། །དྲི་མེད་འོད་གསལ་རྡོ་རྗེ་སྙིང་པོའི་བསྟན། །ཕྱོགས་མཐར་ཁྱབ་ཅིང་སྤེལ་བའི་ཕྲིན་ལས་གྲོགས། །ཞལ་བཞེས་ཐ་ཚིག་གཉན་ལ་མ་འདའ་བར། །མཆོད་གཏོར་བཞེས་ལ་བཅོལ་བའི་ཕྲིན་ལས་མཛོད། །དེ་འོག་ལྗོངས་འདིར་གཟུགས་སྐུའི་རྫུ་འཕྲུལ་གྱིས། །རིགས་སྲས་གདུང་རྒྱུད་སྤེལ་ཞིང་འགྲོ་དོན་སྐྱོངས། །མཉམ་མེད་འཇིག་རྟེན་མགོན་པོའི་བཀའ་བསྒོས་བཞིན། །བཀའ་བརྒྱུད་བསྟན་པ་སྲུང་བར་མ་ཆད་དམ། །མ་འོངས་རྗེས་འཇུག་སྐྱོབ་པར་མ་ཆད་དམ། །ཤ་ཁྲག་བརྒྱན་པའི་གཏོར་མ་འདི་བཞེས་ལ། །ཕྱི་ནང་བར་ཆད་རྐྱེན་ཟློགས་འགལ་རྐྱེན་སོལ། །རྣལ་འབྱོར་བཅོལ་བའི་ཕྲིན་ལས་བསྒྲུབ་པར་མཛོད། །ད་ལྟ་སྒྲུབ་པའི་བརྒྱུད་འཛིན་རྣལ་འབྱོར་གྱིས། །རྩ་གསུམ་དཀྱིལ་འཁོར་ཞལ་ཕྱེས་མཁའ་འགྲོའི་ཚོགས། །བསྙེན་ཅིང་བརྟེན་ནས་མཆོད་བསྟོད་བསྐུལ་བའི་ཚེ། །ཇི་ལྟར་དམ་ཚིག་བརྩེ་གདུང་འབྲལ་མེད་པར། །རྣལ་འབྱོར་སྲུང་ཞིང་སྐྱོབ་པར་མ་ཆད་དམ། །འཁོར་གྱི་ཕྲིན་ལས་ལྷ་མོ་བཞི་དང་ནི། །ཚེ་རིང་མཆེད་ལྔ་མ་ཚོགས་བྱེ་བ་འབུམ། །གཙོ་མོ་བཀའ་བཞིན་དམ་ཚིག་རྗེས་དྲན་ནས། །སྙིང་པོའི་བསྟན་པ་སྤེལ་དང་སྐྱོང་བ་དང༌། །ལས་བཞིའི་དངོས་གྲུབ་སྩོལ་བར་མ་ཆད་དམ། །དམ་ཉམས་དགྲ་བགེགས་བྱད་མ་རྦོད་གཏོང་དང༌། །ལྟས་ངན་ཆོ་འཕྲུལ་བཟློག་པར་མ་ཆད་དམ། །ཤ་ཁྲག་དམར་གྱི་གཏོར་མ་འདི་ལོངས་ལ། །རྣལ་འབྱོར་བཅོལ་བའི་ཕྲིན་ལས་བསྒྲུབ་པར་མཛོད། །

ཅེས་རོལ་ཆེན་དང་བཅས་གཏོར་མ་ཕུལ་ནས། གཙང་རིས་ཀྱི་དཔལ་མགོན་བདུན་ཅུ་དང༌། ཡུལ་ལྷ་སོགས་ལ་དཀར་གསུམ་ལྡན་པའི་གཏོར་འབུལ་ནི། ཨ་ཀཱ་རོས་བདུད་རྩིར་བྱིན་གྱིས་བརླབས་ལ། ཧཱུྃ་བྷྱོ། སྔོན་ཚེ་པདྨ་སམ་བྷཱ་ཝས། །དབང་བསྐུར་བཀའ་དང་དམ་ཚིག་ཅན། །ཚངས་དབང་བརྒྱ་བྱིན་རྒྱལ་ཆེན་བཞི། །ཕྱོགས་སྐྱོང་བཅུ་དང་ལྷ་ཆེན་བརྒྱད། །ཀླུ་བརྒྱད་གཟའ་བརྒྱད་གནོད་སྦྱིན་བརྒྱད། །རྒྱུ་སྐར་ལྷ་མོ་ཉེར་བརྒྱད་དང༌། །ལྷ་སྲིན་སྡེ་བརྒྱད་གཞི་བདག་དང༌། །གཉུག་མར་གནས་པའི་འབྱུང་པོའི་ཚོགས། །མཆོད་སྦྱིན་གཏོར་མ་འདོད་ཡོན་ལྔ། །བཞེས་ནས་སངས་རྒྱས་བསྟན་པ་སྲུང༌། །རྣལ་འབྱོར་སྐྱོང་ཞིང་ལས་བཞི་དང༌། །ཐུན་མོང་དངོས་གྲུབ་འདོད་དགུའི་གཏེར། །མྱུར་དུ་འགྲུབ་པའི་སྡོང་གྲོགས་མཛོད། །གཞན་ཡང་བཅོལ་བའི་ཕྲིན་ལས་ཀུན། །ཐོགས་མེད་བདེ་བླག་བསྒྲུབ་ཏུ་གསོལ། ། ཞེས་དཀར་གཏོར་ཕུལ། དེ་ནས་བརྟན་མ་སྐྱོང་བ་ནི། བྷྱོ། རྒྱ་གར་བལ་བོད་ཡུལ་ལྗོངས་སུ། །གནས་པའི་སྐྱོང་མ་ཡལ་གམ་ཅན། །རྡོ་རྗེ་ཀུན་གྲགས་ཡ་མ་སྐྱོང༌། །ཀུན་བཟང་མ་དང་བགེགས་ཀྱི་གཙོ། །རྡོ་རྗེ་དཔལ་ལྡན་ཁྱུང་བཙུན་མ། །དྲག་མོ་རྒྱལ་དང་མཁའ་ལྡིང་མ། །ཀོང་བཙུན་རྡོ་རྗེ་སྨན་གྱི་ཡུམ། །གཡར་མོ་བསིལ་དང་ཟུ་ལེ་མ། །བདུད་མོ་གནོད་སྦྱིན་སྨན་མོ་སྟེ། །སྔོན་ཚེ་མཆིམས་ཕུའི་དབེན་གནས་སུ། །སློབ་དཔོན་པདྨ་ཐོད་ཕྲེང་དང༌། །གླང་རི་དཔལ་གྱི་སེངྒེ་ཡིས། །དབང་བསྐུར་དཀྱིལ་འཁོར་མཐའ་སྐྱོང་བར། །ཁས་བླངས་ཐ་ཚིག་དྲན་པར་གྱིས། །བཀའ་ཡི་གཏོར་མ་འདི་ལོངས་ལ། །བསམ་པ་སྒྲུབས་ལ་ཚེ་བསོད་སྤེལ། །བར་ཆད་རྐྱེན་ཟློགས་ནད་ཡམས་ཀྱི། །དལ་ཁ་བསྒྱུར་ལ་མཚོན་རྩེ་ཐུལ། །བཅོལ་བའི་ཕྲིན་ལས་བསྒྲུབ་པར་མཛོད། །མ་མ་ཧྲིང་ཧྲིང་བ་ལིཾ་ཏ་ཁ་ཁ་ཁཱ་ཧི། ཞེས་གཏོར་མ་དང་བཤལ་ཆུ་ཕུལ། དེ་ནས་རྗེས་ཀྱི་རིམ་པ་ལ། གཏང་རག་གི་མཆོད་པ་འབུལ་བ་ནི། མཆོད་རྫས་རྣམས་གསོ་བྲན་ལེགས་པར་བྱས་ཏེ།

བྷྲཱུྃ་ལས་རིན་ཆེན་སྣོད་རྣམས་སུ། །མཆོད་རྫས་ཡིད་འོང་ཕུན་སུམ་ཚོགས། །ཨོཾ་ཨཱཿཧཱུྃ་གིས་བྱིན་བརླབས་ནས། །ཟག་མེད་བདེ་ཆེན་སྐྱེད་བྱེད་གྱུར། །ཧོ། བླ་མ་ཡི་དམ་ཆོས་སྐྱོང་དང༌། །མ་གཅིག་མཁའ་འགྲོ་འཁོར་བཅས་ལ། །གཙང་བྱེད་མཆོད་ཡོན་ཞབས་བསིལ་དང༌། །ཡིད་འོང་མེ་ཏོག་བདུག་པའི་སྤོས། །རྨད་བྱུང་སྣང་གསལ་དྲི་ཆབ་གཏམས། །ངོ་མཚར་ཞལ་ཟས་རོལ་མོ་ལྡིར། །གཟུགས་སྒྲ་དྲི་རོ་རེག་བྱ་སྟེ། །གསང་གསུམ་དགྱེས་པ་བསྐྱེད་པའི་ཕྱིར༌། །སྨན་དང་གཏོར་མ་རཀྟའི་ཚོགས། །སྣང་སྲིད་ཡོངས་བསྐང་མཆོད་པའི་སྤྲིན། །འབུལ་གྱི་ཐུགས་བརྩེར་དགྱེས་དགོངས་ལ། །བཞེས་ནས་དངོས་གྲུབ་སྩལ་དུ་གསོལ། །ཨོཾ་ཤྲཱི་མན་གུ་རུ་དེ་ཝ་དྷརྨ་པཱ་ལ་ཌཱ་ཀི་ནི་ས་པ་རི་ཝཱ་ར་བཛྲ་ཨརྒྷཾ། །མ་གཅིག་བརྟེན་ཅིང་སྒྲུབ་པའི་རྣལ་འབྱོར། །ཚེ་མཐར་ཕྱིན་ཅིང་དགེ་ཚོགས་ཇི་སྙེད། །ཡིད་བཞིན་ལྷུན་གྲུབ་དཔལ་དང་འབྱོར་རྒྱས། །སྙན་གྲགས་དབང་ཐང་བསོད་ནམས་ཡེ་ཤེས། །མི་ནུབ་རྒྱལ་མཚན་བཞིན་དུ་བསྒྲེང་བའི། །དམ་ཚིག་བཀའ་ལས་མ་འདའ་ཚུལ་བཞིན། །སྒྲུབས་ཤིག་སྤེལ་ཅིག་བཅོལ་བའི་ཕྲིན་ལས། །ད་ལྟ་ཉིད་དུ་འགྲུབ་པར་མཛོད་ཅིག ། ཅེས་རིན་ཆེན་བདུད་རྩིའི་སྐྱེམས་ནམ་མཁའི་དབྱིངས་སུ་མཆོད་ནས། བསྟོད་པ་ནི། ཧྲཱིཿ སྒྲུབ་བརྒྱུད་བསྟན་སྲུང་ཕྲིན་ལས་མྱུར་མགྱོགས་མ། །རྡོ་རྗེ་ཅན་གྱི་བཀའ་དང་དམ་ཚིག་ཅན། །མཆོག་རྣམས་བསྐྱེད་པའི་རྒྱལ་ཡུམ་འཕགས་མ་སྟེ། །ཨ་ཕྱི་ཆོས་ཀྱི་སྒྲོལ་མར་ཕྱག་འཚལ་བསྟོད། །གཙོ་མོ་ལས་སྤྲུལ་མཁའ་འགྲོ་ཆེན་མོ་བཞི། །གཞན་ཡང་སྤྲུལ་པ་ཡང་འཁོར་ཚོགས་རྣམས་བསྟོད། །གྲུབ་པའི་དགྱེས་སྡེ་སྨན་བཙུན་དཔལ་གྱི་ཡུམ། །ཚེ་རིང་མཆེད་ལྔ་བྲན་གཡོག་དམ་ཅན་ཚོགས། །དང་བའི་ཡིད་ཀྱིས་རབ་ཏུ་བསྟོད་པར་བགྱི། ། ཞེས་མཆོད་དང་བསྟོད་པས་འཁོར་བཅས་རབ་ཏུ་དགྱེས་ཤིང་ཚིམ་པར་མོས་ལ། ཕྱྭ་གཡང་འགུག་པའི་འདོད་གསོལ་སོགས་སྤྲོ་ན་འདིར་བྱ་ཞིང༌། དེ་མིན་མཆོད་ཕྲེང་ལྟ་བུ་ལ་སྐབས་འདིར་དངོས་གྲུབ་བླང་བའི་དམིགས་པ་ནི། རྩ་བ་གསུམ་གྱི་སྐུ་ལས་དངོས་གྲུབ་ཀྱི་ཆར་དང༌། འཁོར་རྣམས་ཀྱི་ཐུགས་ཀ་ནས་འོད་ཟེར་དངོས་གྲུབ་ཀྱི་ཚུལ་དུ་གཙོ་མོ་ལ་ཐིམ། དེ་ཉིད་ཀྱི་སྐུ་ཡི་ཆ་ནས་མི་ཟད་པའི་ལོངས་སྤྱོད་འདོད་ཡོན་ཆར་བཞིན་བབས་ནས་རང་ཉིད་དང་རྟེན་རྫས་རྣམས་ལ་ཐིམ་པར་བསྒོམས་ཤིང༌། ཧྲཱིཿ མ་གཅིག་མཁའ་འགྲོ་ཆེན་མོ་ཡིས། །དངོས་གྲུབ་སྩོལ་བའི་དུས་ལ་བབས། །རྩ་གསུམ་དཀྱིལ་འཁོར་ལྷ་ཚོགས་དང༌། །འཁོར་གྱི་ཌཱ་ཀིའི་ཐུགས་ཀ་ནས། །འོད་ཟེར་འཕྲོས་ནས་གཙོ་མོར་ཐིམ། །སྐུ་ཡི་ཆ་ནས་མི་ཟད་པའི། །རིན་ཆེན་ནོར་བུ་འདོད་དགུའི་ཆར། །ལྷ་ཀླུ་མི་ཡི་ལོངས་སྤྱོད་བསྡུས། །བདག་དང་སྒྲུབ་རྫས་རྣམས་ལ་ཐིམ། །དཔལ་དང་འབྱོར་པ་ལྷུན་གྲུབ་གྱུར། །ཨོཾ་ཌཱ་ཀི་ནི་སརྦ་མ་ཎི་རཏྣ་སིདྡྷི་ཕ་ལ་ཧཱུྃ།

གསུམ་མམ་ཉེར་གཅིག་བཟླའོ། །དེ་ནས་ནོངས་པ་བཟོད་གསོལ་ནི། ཧོ། དབྱིངས་ཕྱུག་སྤྲུལ་པའི་ཡུམ་ཆེན་འཁོར་བཅས་དགོངས། །སྣང་སེམས་གདོད་ནས་ཀུན་བཟང་རོལ་པ་ལ། །གཉིས་འཛིན་འཁྲུལ་པའི་དྲི་མས་བསླད་པ་ཡི། །སྐུ་ཡི་དམ་ཚིག་ཉམས་པ་བཟོད་པར་གསོལ། །གྲག་སྟོང་ངོ་བོ་གསུང་གི་ཆ་ཤས་ལས། །སྙིང་པོ་སྔགས་ཀྱི་ཛབ་བཟླས་སྒྲ་ཟུར་ཆག །བག་མེད་ཉེས་པའི་རྡུལ་ཕྲན་ཆ་བསགས་པ། །གསུང་གི་དམ་ཚིག་ཉམས་པ་བཟོད་པར་གསོལ། །ལྷ་བསྐྱེད་ཟབ་མོའི་ཏིང་འཛིན་གཡེལ་བ་དང༌། །སྤྲོ་བསྡུའི་དམིགས་པ་འཐིབ་ཅིང་དོན་མ་རྟོགས། །མ་རིག་ལོག་རྟོག་ཉོན་མོངས་འཕྲོ་རྒོད་ཀྱིས། །ཐུགས་ཀྱི་དམ་ཚིག་ཉམས་པ་བཟོད་པར་གསོལ། །བླ་མ་ལྷ་དང་རྡོ་རྗེ་སྤུན་གྲོགས་དང༌། །ཕ་མར་བརྙས་ཤིང་སྡང་སེམས་ཚིག་བརླང་སོགས། །སྣོད་མིན་ཉམས་པ་ཅན་ལ་གསང་བ་བསྒྲགས། །མཚམས་མེད་ལས་བགྱིས་ཐམས་ཅད་བཟོད་པར་གསོལ། །གཞན་ཡང་མཆོད་གཏོར་མ་ཚང་ཕུད་ཉམས་པ། །རྒྱུད་དང་དུས་ཀྱི་གཏོར་མ་ལོ་ཟླར་འགྱང༌། །ཚུལ་བཞིན་མི་ལྡན་ལེ་ལོའི་དབང་གྱུར་པའི། །ནོངས་པ་ཅི་མཆིས་བཤགས་སོ་དག་ཚངས་སྩོལ། །ཨོཾ་བཛྲ་ས་ཏྭ་ས་མ་ཡ་ སོགས་ཡིག་བརྒྱ་ལན་གསུམ་བརྗོད། ཆད་ལྷག་བསྐང་ནས་འདོད་དོན་གསོལ་བ་ནི། ཧྲཱིཿ བདེ་ཆེན་དགའ་སྟེར་རྒྱལ་ཀུན་ཡུམ་གྱུར་མ། །རྡོ་རྗེ་མཁའ་འགྲོ་ཆེན་མོ་འཁོར་བཅས་རྣམས། །བསྟེན་ཅིང བསྒྲུབ་དང་མཆོད་བསྟོད་བསྐུལ་བའི་མཐུས། །བསྟན་པ་དར་རྒྱས་འགྲོ་ལ་ཕན་བདེ་སྤེལ། །རྣལ་འབྱོར་བསམ་དོན་ལྷུན་གྱིས་གྲུབ་པ་དང༌། །བླ་མེད་བྱང་ཆུབ་ཆེན་པོ་མངོན་གྱུར་ནས། །ཨུ་རྒྱན་མཁའ་སྤྱོད་བཅོམ་ལྡན་རྣམ་སྣང་མའི། །དབྱིངས་དང་ཡེ་ཤེས་དབྱེར་མེད་རོ་གཅིག་ཤོག ། ཅེས་སྨོན་ལམ་བཏབ་ལ། དེ་ནས་མདུན་མཁར་བསྐྱེད་པའི་རྩ་གསུམ་རྣམས་རང་ལ་ཐིམ་པར་མོས་ཤིང༌། ཧོ། རྩ་གསུམ་རྒྱལ་བ་རྒྱ་མཚོའི་ཚོགས། །འོད་ཟེར་དཀར་དམར་མཐིང་ཀ་དང༌། །འབྲུ་གསུམ་ངོ་བོར་འོད་ཞུ་ནས། །རང་ཉིད་ལྷར་གསལ་གནས་གསུམ་ལ། །ཐིམ་པས་སྒོ་གསུམ་སྒྲིབ་དག་ནས། །སྐུ་གསུམ་འབྲས་བུ་མངོན་དུ་གྱུར། །ཨོཾ་ཨཱཿཧཱུྃ།

ཞེས་བརྗོད་ཅིང༌། མཆོད་སྦྱིན་གནས་འདིར་གང་ལྷགས་པའི། ། སོགས་ཀྱིས་འཇིག་རྟེན་པ་རྣམས་རང་གནས་སུ་གཤེགས་ཤིང༌། བསྔོ་བ་སྨོན་ལམ་ཡང་དཀྱུས་ལྟར་བྱའོ། །རྗེས་ནི། བདག་ཉིད་བསྐྱེད་པའི་དཀྱིལ་འཁོར་པ། །རིམ་བཞིན་འོད་ཞུ་ཧྲཱིཿལ་ཐིམ། །དེ་ཡང་དམིགས་མེད་བློ་འདས་ཀྱི། །ཁྱབ་བདལ་ཆེན་པོའི་ཀློང་དུ་བཞག ། ཅེས་དུས་གསུམ་སྤྲོས་པ་དང་བྲལ་བ་རང་གསལ་བརྗོད་མེད་མ་བཅོས་རྫོགས་པ་ཆེན་པོའི་ངང་དུ་མཉམ་པར་བཞག་ནས། ཅི་སྣང་ཅི་ཤར་གྱི་ཆོས་ཉིད་ཐམས་ཅད་ཡེ་ཤེས་ལྔ་ཡི་རོལ་པར་སྤྱོད་པར་བྱའོ།། །། ༈ དེ་ལྟར་རྒྱལ་ཡུམ་མཁའ་འགྲོའི་གཙོ་མོ་འདི་ཉིད་ཀྱི་ཆེ་བའི་ཡོན་ཏན་ཟུར་ཙམ་བརྗོད་ན། ཕྱི་ལྟར་ན་མདོ་སྔགས་ཀྱི་ཚད་མར་གྱུར་བའི་རིང་ལུགས་ཡོངས་དང༌། ཁྱད་པར་སྒྲུབ་བརྒྱུད་སྙིང་པོའི་བསྟན་པ་རིན་པོ་ཆེའི་བཀའ་སྲུང་གཙོ་མོར་བསྔགས་ཤིང༌། ནང་ལྟར་ན་ཆོས་དབྱིངས་བདེ་བ་ཆེན་པོའི་ཀློང་དུ་ཀུན་ཏུ་བཟང་མོའི་རོལ་པ་སྟེ་རྡོ་རྗེ་རིགས་ཀྱི་མཁའ་འགྲོ། གསང་བ་ལྟར་ན་ཨོ་རྒྱན་རང་བྱུང་གནྡྷོ་ལར་ཌཱ་ཀིའི་ཚོགས་དཔོན་ཆེན་མོ་སྟེ། ཡེ་ཤེས་ལས་དང་འཇིག་རྟེན་གྱི་སྤྲུལ་བསྒྱུར་དུ་མས་འགྲོ་བ་རྣམས་ལེགས་པའི་ལམ་ལ་བཀོད་ཅིང༌། མཆོག་དང་ཐུན་མོང་གི་དངོས་གྲུབ་སྟེར་བ་དང༌། ཚེ་དཔལ་སྤེལ་ཅིང འདོད་དགུ་དབང་དུ་བསྒྱུར་བ་དང༌། བསྟན་དགྲ་ཐལ་བར་བརླག་པ་སོགས་དམ་ཚིག་ཅན་ལ་སྙིང་ཉེ་ཞིང༌། བཀྲ་ཤིས་ལ་བྱིན་ཆགས་པ། ཕྲིན་ལས་རྣོ་ལ་ཐུགས་རྗེ་མྱུར་བ། མཐར་ཐུག་ཨོ་རྒྱན་མཁའ་སྤྱོད་ཀྱི་ཞིང་དུ་འདྲེན་པར་ནུས་པའི་མཐུ་དང་བྱིན་རླབས་ཆེས་ཁྱད་པར་འཕགས་པ་དང༌། མཉམ་མེད་འབྲི་གུང་བ་ཆེན་པོ་འཇིག་རྟེན་གསུམ་གྱི་མགོན་པོ་མཆོག་ལ་སོགས་པ་དང༌། །གར་དབང་རྡོ་རྗེ་འཆང་ཆེན་དྷརྨཱ་ཤྭ་ར་སོགས་བསྟན་པའི་བདག་པོ་རྣམས་ཀྱང༌། ཀྱུ་རའི་གདུང་ཞེས་ཆོས་དང་སྲིད་ཀྱི་དབང་འབྱོར་པ། སྣ་ནམ་བཟའ་ཨ་ཕྱི་ཆོས་ཀྱི་སྒྲོལ་མའི་རིགས་སྲས་ཀྱི་གདུང་བརྒྱུད་དྲི་མེད་ལས་འབྱོན་པར་གྲགས་པའི།

དེང་གི་བར་གྲུབ་བརྙེས་གོང་མ་རྣམས་ཀྱི་བཀའ་དམ་ལ་གནས་པའི་རིགས་ཆོས་རྒྱུད་ཀྱི་སྲུང་མ་ཐུན་མོང་མ་ཡིན་པ་མངོན་སུམ་ཚད་མར་གྲུབ་ཅིང༌། ཞི་སྒྲུབ་ཕལ་ཆེར་ལ་གཙོ་འཁོར་བཞེངས་སྟབས་ཅན་ལས་ཞོན་པ་སོགས་མེད་ཀྱང༌། འདིར་ནི་ཕྲིན་ལས་མྱུར་མགྱོགས་རྩལ་ལྡན་གྱི་རྟེན་འབྲེལ་ཆེད་དུ་བསྔགས་ཤིང༌། འཁོར་དུ་ཚེ་རིང་མཆེད་ལྔ་བསྐྱེད་པའང་དཔའ་དང་དཔུང་གི་ཚུལ་གྱི་དགོས་པ་ཁྱད་པར་ཅན་དུ། སྔོན་གྱི་རྗེ་བཙུན་གོང་མ་རྣམས་ཀྱི་རྡོ་རྗེའི་བཀའ་གསང་ལས་བྱུང་བ་དང༌། འདི་ཉིད་བསྙེན་ཅིང་བརྟེན་པའི་རྣལ་འབྱོར་པས་བྱམས་སྙིང་རྗེ་སེམས་བསྐྱེད་དམ་ཚིག་དང་ལྡན་པའི་སྒོ་ནས་བསྒྲུབ་ན་ཇི་བཞིན་ངེས་ཤེས་སྐྱེ་བའི་རྟགས་མཚན་ངོ་མཚར་ཅན་འཆར་བ་ཡིན་ནོ། །དེ་ལྟར་བསྟན་པའི་བཀའ་སྲུང་ཆེན་མོའི་མཆོད་ཕྲེང་འདི་ཡང་དབོན་སྤྲུལ་ཀརྨ་འཆི་མེད་ཀྱིས་བསྐུལ་བར་བརྟེན། སི་ཏུ་སྤྲུལ་སྐུ་པདྨ་ཉིན་བྱེད་དབང་པོས་དཔལ་སྤུངས་ཆོས་འཁོར་གྱི་ཡང་དབེན་དུ་སྤེལ་བ་བསྟན་འགྲོ་ཕན་བདེའི་དགེ་ལེགས་ཀྱིས་ཁྱབ་པར་གྱུར་ཅིག །སརྦ་མངྒ་ལམ། ། ༈ མཆོད་ཕྲེང་བཀྲ་ཤིས་ཆར་འབེབས་ཀྱི་གུད་སྦས་ནོར་སྒྲུབ་བཞུགས་སོ། ། ན་མོ་པདྨཱ་ཀ་ར་དྷེ་ཝ་ཌཱ་ཀི་ནི་ཡེ། རྩ་གསུམ་ལྷ་ཚོགས་བཀའ་ཡི་གནང་བ་སྩོལ། འདིར་བསྟན་པའི་བཀའ་སྲུང་ཆེན་མོ་རིག་འཛིན་རྒྱུད་ཀྱི་སྲུང་མ་ཆོས་སྐྱོང་ཕྲིན་ལས་མྱུར་མགྱོགས་ཨ་ཕྱི་ཆེན་མོའི་སྒོ་ནས་ཚེ་གཡང་དངོས་གྲུབ་འགུགས་པའི་རིམ་པ་ལ། མཆོད་གཏོར་བཤམས་བཀོད་སོགས་གཞུང་གིས་འཐུས་ཤིང༌། ཐོད་པ་མཚན་ལྡན་གྱི་ནང་དུ་རིན་པོ་ཆེ་སྣ་ལྔ་དང་འབྲུ་སྣ་སྨན་སྣ་སོགས་ཀྱིས་བསྐང་བ་དང༌། མདའ་རྒོད་སྒྲོ་མ་ལ། དངུལ་གྱི་མེ་ལོང༌། རིན་ཆེན་སྣ་ལྔ། དར་སྣ་སྟག་རས་ཀྱིས་བརྒྱན་པ་དང༌། གཞན་ཡང་གཡང་རྫས་གང་འབྱོར་སྟེགས་ཀྱི་ཁར་བཤམས་ལ། སྔོན་དུ་རང་ཉིད་ལྷ་ཡི་ང་རྒྱལ་གྱི་ཏིང་ངེ་འཛིན་དང་ལྡན་པའི་སྒོ་ནས། མཁའ་འགྲོ་འཁོར་བཅས་ཀྱི་བསྐྱེད་བཟླས་མཆོད་ཕྲེང་ཚོགས་དང་སྐོང་བཤགས་སོགས་དང༌།

བསྐུལ་ཆད་ཐོ་བརྟན་སྐྱོང་གྲུབ་ནས། གཏང་རག་མཆོད་འབུལ་དང༌། གསོལ་ཀ་དང་བསྟོད་པ་ཚར་ནས་དངོས་གྲུབ་ལེན་ཆོག་དང་སྟབས་གཅིག་འདིར་འཇུག་པར་བྱ་སྟེ། མ་ཡེངས་ངང་ནས་མདའ་དར་གཡབ་ཅིང་བདུག་སྤོས་བསྲེགས་ལ་འདི་སྐད་དོ། ། ཨོཾ། མ་གཅིག་ཡོན་ཏན་ཀུན་འབྱུང་གནས། །དཔག་བསམ་ཡོངས་འདུའི་ལྗོན་བཟང་བཞིན། །ཡིད་ལ་འདོད་འཇོའི་དཔལ་སྟེར་མ། །ཁྱོད་ཀྱི་ཐུགས་དམ་དབྱིངས་ནས་བསྐུལ། །བདག་ཅག་རྣལ་འབྱོར་སྒྲུབ་པ་པོས། །མ་ཁྱོད་མཆོད་ཅིང་བསྟེན་པའི་མཐུས། །དངོས་གྲུབ་སྩོལ་བའི་དུས་ལ་བབས། །སྐུ་ཡི་དཀྱིལ་འཁོར་བསྟན་དུ་གསོལ། །ཐུགས་རྗེའི་ཐུགས་དམ་མ་གཡེལ་ཅིག །བརྩེ་བ་ཆེན་པོས་རྗེས་དགོངས་ནས། །དཔལ་འབར་གཙོ་མོ་ཡུམ་གྱུར་མའི། །ཐུགས་ཀའི་ས་བོན་ལས་འཕྲོས་པའི། །ཕྱྭ་གཡང་ནོར་བདག་དབང་ཕྱུག་མ། །རྒྱས་བྱེད་ལྷ་མོ་གསེར་མདོག་ཅན། །ཞི་འཛུམ་ཞལ་གཅིག་ཕྱག་གཉིས་གཡས། །ནེའུ་ལེ་གཡོན་པས་ཀ་པཱ་ལ། །ཡིད་བཞིན་ནོར་བུས་ཕྱུར་བུར་བཀང༌། །ཟ་འོག་བེར་སེར་ལྷབ་ལྷུབ་གསོལ། །འཁྱུགས་ལྡེམ་གར་སྟབས་ལིངས་སེ་ལིང༌། །རིན་ཆེན་སྣ་ཚོགས་རྒྱན་ལྡན་གསལ། །མདུན་བསྐྱེད་རྩ་གསུམ་སྐུ་ལས་ཀྱང༌། །འོད་འཕྲོས་ཚེ་ལྷ་རིགས་ལྔ་དང༌། །སྒྲོལ་མ་ནོར་རྒྱུན་ལྷ་མོའི་ཚོགས། །རྣམ་སྲས་རྨུགས་འཛིན་ཛམ་ལྷའི་རིགས། །སྐུ་དང་ཕྱག་མཚན་དངོས་གྲུབ་ཀྱི། །འདོད་དགུའི་ལོངས་སྤྱོད་སྤྲིན་ལྟར་གཏིབས། །རྒྱས་བྱེད་དཔལ་མོར་གཉིས་མེད་ཐིམ། །འཁོར་ཚོགས་མཁའ་འགྲོ་མ་ཀུན་གྱི། །ཐུགས་ལས་འོད་ཟེར་སྣ་ཚོགས་དང༌། །ཕྱྭ་གཡང་དངོས་གྲུབ་ཀུན་བསྡུས་ནས། །སྤྲུལ་པའི་ལྷ་མོར་ཐིམ་པར་གྱུར། །དེ་ལས་དངོས་གྲུབ་ནམ་མཁའི་མཛོད། །ཇི་བཞིན་རྒྱ་ཆེན་མཐའ་ཀླས་པ། །ལྷ་ཀླུ་མི་ཡི་ཕྱྭ་གཡང་གི །ཚེ་བསོད་དཔལ་གྲགས་ལོངས་སྤྱོད་ཀྱི། །དྭངས་བཅུད་ཡིད་བཞིན་ནོར་བུའི་ཆར། །ཕྱྭ་དང་གཡང་གི་དངོས་གྲུབ་ནི། །ཉི་མ་བཞིན་དུ་ཀྱི་ལི་ལི། །ཟླ་བ་བཞིན་དུ་སྒ་ར་ར། །སྐར་ཚོགས་བཞིན་དུ་སྤུངས་སེ་སྤུངས། །བར་མཚམས་མེད་པར་གང་བར་གྱུར། །

ཅེས་དམིགས་ནས། ཨོཾ་ཛྙ་ན་ཌཱ་ཀི་ནི་ཡེ་ཧ་རི་ནི་ས་ཧཱུྃ་ཕཊ། ཨོཾ་གྷ་མུ་ཨ་ཙི་ཏ་ཤྭ་ར་དྷརྨ་ཛྙ་ན་ས་མ་ཡ་ཏ་ར་དི་མ་ཌཱ་ཀི་ནི་ཡེ། སརྦ་མ་ཎི་རཏྣ་སིདྡྷི་ཕ་ལ་ཧཱུྃ། སྔར་བཞིན་མདུན་བསྐྱེད་གཙོ་བོའི་ཐུགས་ཀའི་སྔགས་འཁོར་ལ་དམིགས་ནས་བཟླས་པ་བྱའོ། །རྩ་བ་གསུམ་གྱི་སྐུ་ལས་དངོས་གྲུབ་ཀྱི་ཆར་དང༌། མདུན་བསྐྱེད་གཙོ་འཁོར་རྣམས་དང༌། སྤྲུལ་པའི་ལྷ་མོའི་སྐུ་ཡི་ཆ་ནས་ཕྱྭ་གཡང་ཟས་ནོར་ཐམས་ཅད་སིབ་སིབ་བབས་ནས་གནས་ལུས་ལོངས་སྤྱོད་དཀོར་ནོར་སོགས་ལ་ཐིམ་ཞིང་ཆིལ་གྱིས་གང་བར་དམིགས་ཏེ་སྔགས་བཟླའོ། །སླར་རང་གི་ཐུགས་ཀ་ནས་མཆོད་པའི་ཚོགས་སྤྲོ་བར་དམིགས་ནས། ཨོཾ་དྷརྨ་ཏཱ་ར་རཏྣ་ཌཱ་ཀི་ནི་ཡེ་བཛྲ་ཨརྒྷཾ། པདྱཾ། པུཥྤེ། དྷཱུ་པེ། ཨ་ལོ་ཀེ། གནྡྷེ། ནཻ་ཝི་དྱ། ཤབྡ་པཱུ་ཛ་ཧོ་ ཡིས་མཆོད་ནས་བསྟོད་པ་ནི། ཧྲཱིཿ རཏྣ་ནོར་གྱི་དཔལ་གཡང་ཅན། །རྒྱས་བྱེད་ལྷ་མོ་དབང་ཕྱུག་མ། །རིན་ཆེན་མཁའ་འགྲོ་གཏེར་བདག་ཡུམ། །ཡིད་བཞིན་ལྷ་མོར་ཕྱག་འཚལ་བསྟོད། །ཕྱོགས་བཅུའི་རྒྱལ་དང་རྩ་གསུམ་གྱི། །ཐུགས་རྗེ་བསྐུལ་བའི་མཐུ་སྟོབས་ལས། །འཆི་མེད་ཚེ་མཐར་ཕྱིན་པ་དང༌། །འཛད་མེད་ལོངས་སྤྱོད་ཐོབ་པར་གྱུར། ། ཅེས་བརྗོད་ནས། །ལྷ་མོ་དེ་ཉིད་ཞི་འཛུམ་དགྱེས་པའི་མདངས་དང་མཚར་སྡུག་འོད་དང་ལྡན་པའི་ངང་ནས་སྐུ་རྟེན་ནམ། གཞན་ཡང༌། རྟེན་རྫས་ཁྱད་པར་ཅན་སོགས་ལ་ཡང་ཡང་ཐིམ་པར་བསམ། སླར་གྲངས་སོགས་གསོག་ན། མཆོད་པ་གསར་དུ་བརྗེ་བའམ། ཁ་གསོ་བ་དང༌། གསེར་སྐྱེམས་ཕུད་མཆོད་ཀྱང་བཤམས་ལ། གོང་གི་གཏང་རག་མཆོད་པ་དེ་དང༌། གསོལ་ཀ་བཅས་དང༌། བསྟོད་པ་གྲུབ་མཚམས་གཡང་འགུགས་དམིགས་པ་སྔགས་བཟླ་བ་སོགས་ཀྱིས་ཐུན་ཚོད་བཟུང་ཞིང༌། །མཐར་ཐུན་རྗེས་མ་ནི་གཞུང་གི་དངོས་གྲུབ་ལེན་པ་དེ་བཞག་ནས། དེ་མིན་ནོངས་པ་བཟོད་གསོལ་སོགས་ལ་དཀྱུས་ལྟར་འཇུག་པར་བྱའོ། །དེ་ལྟར་ཡེ་ཤེས་མཁའ་འགྲོའི་ལས་ཚོགས་ཀྱི། །ཚེ་གཡང་འགུགས་པའི་ཐབས་གནད་སྤེལ་བ་ལ། །ནོངས་པ་མཆིས་ན་ཌཱ་ཀིས་བཟོད་པར་མཛོད། །དགེ་བས་བསྟན་འགྲོར་བཀྲ་ཤིས་ཁྱབ་གྱུར་ཅིག
